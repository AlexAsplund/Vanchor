vanchor.analysis

Simulation analysis framework: run scenarios, record full time series, and turn them into metrics, terminal reports, CSV, and plots.

Quick start::

from vanchor.analysis import run_scenario, anchor_metrics, print_report
from vanchor.analysis.scenarios import SCENARIOS

log = run_scenario(SCENARIOS["anchor_tight"])
print_report(log, anchor_metrics(log))

Or from the shell::

python -m vanchor.analysis anchor_tight --plot out/anchor.png --csv out/anchor.csv

def run_scenario( scenario: Scenario) -> SimLog: 
135def run_scenario(scenario: Scenario) -> SimLog:
136    """Run a scenario deterministically and return its recorded :class:`SimLog`."""
137    sim = Simulator(
138        start=BoatState(point=scenario.start, heading_deg=0.0),
139        environment=scenario.environment,
140        model=scenario.model,
141    )
142    state = NavigationState()
143    state.max_steer_angle_deg = scenario.max_steer_angle_deg
144    nav = Navigator(state, bus=None)
145    controller = Controller(
146        state,
147        sim.motor,
148        bus=None,
149        tick_hz=scenario.control_hz,
150        helm=Helm(scenario.helm_pid) if scenario.helm_pid else None,
151        anchor_config=scenario.anchor_config,
152        waypoint_config=scenario.waypoint_config,
153        drift_config=scenario.drift_config,
154        cruise_pid=scenario.cruise_pid,
155    )
156    gps = SimGps(sim.truth, bus=None, update_hz=scenario.gps_hz)
157    compass = SimCompass(sim.truth, bus=None, update_hz=scenario.compass_hz)
158
159    # Prime perceived state with one fix + heading.
160    nav.handle_sentence(gps.sample(sim.truth()))
161    nav.handle_sentence(compass.sample(sim.truth()))
162
163    pending = sorted(scenario.commands, key=lambda c: c.t)
164    dt = scenario.physics_dt
165    gps_period = 1.0 / scenario.gps_hz
166    compass_period = 1.0 / scenario.compass_hz
167    ctrl_period = 1.0 / scenario.control_hz
168
169    t = 0.0
170    next_gps = next_compass = next_ctrl = 0.0
171    samples: list[Sample] = []
172
173    while t < scenario.duration_s:
174        # Fire any commands due at or before now.
175        while pending and pending[0].t <= t:
176            _dispatch(state, sim, controller, pending.pop(0).command)
177
178        sim.step(dt)
179        if t >= next_gps:
180            nav.handle_sentence(gps.sample(sim.truth()))
181            next_gps += gps_period
182        if t >= next_compass:
183            nav.handle_sentence(compass.sample(sim.truth()))
184            next_compass += compass_period
185        if t >= next_ctrl:
186            controller.control_tick(ctrl_period)
187            next_ctrl += ctrl_period
188
189        samples.append(_record(t, state, sim))
190        t += dt
191
192    return SimLog(scenario, samples)

def anchor_metrics( log: SimLog, *, tail_seconds: float = 30.0, settle_tolerance_m: float | None = None) -> AnchorMetrics: 
 93def anchor_metrics(
 94    log: SimLog, *, tail_seconds: float = 30.0, settle_tolerance_m: float | None = None
 95) -> AnchorMetrics:
 96    """Analyse a station-keeping run from its ground-truth distance to anchor."""
 97    samples = [s for s in log.samples if not math.isnan(s.dist_anchor_truth_m)]
 98    if not samples:
 99        raise ValueError("no anchor set during this run -- nothing to analyse")
100
101    ts = [s.t for s in samples]
102    dist = [s.dist_anchor_truth_m for s in samples]
103    thrust = [s.thrust for s in samples]
104    radius = samples[-1].anchor_radius_m
105    tol = settle_tolerance_m if settle_tolerance_m is not None else max(radius, 1.5)
106
107    start_distance = dist[0]
108    closest_i = min(range(len(dist)), key=lambda i: dist[i])
109    closest = dist[closest_i]
110    # Overshoot = how far it swung back out *after* its closest approach.
111    after = dist[closest_i:]
112    overshoot = (max(after) if after else closest)
113
114    # Settling time: last moment it was still outside tolerance (it "settles"
115    # once it never again leaves the tolerance band).
116    settling_time = math.nan
117    for i in range(len(dist)):
118        if dist[i] <= tol and all(d <= tol for d in dist[i:]):
119            settling_time = ts[i] - ts[0]
120            break
121
122    # Post-settle / tail-window stats.
123    tail = [s for s in samples if s.t >= ts[-1] - tail_seconds]
124    tail_d = [s.dist_anchor_truth_m for s in tail]
125    within = (
126        100.0 * sum(1 for d in tail_d if d <= radius) / len(tail_d) if tail_d else 0.0
127    )
128
129    # Thrust usage.
130    reversals = 0
131    for a, b in zip(thrust, thrust[1:]):
132        if (a > 0.02 and b < -0.02) or (a < -0.02 and b > 0.02):
133            reversals += 1
134    effort = sum(abs(b - a) for a, b in zip(thrust, thrust[1:]))
135    rev_frac = sum(1 for x in thrust if x < -0.02) / len(thrust)
136
137    return AnchorMetrics(
138        radius_m=radius,
139        start_distance_m=start_distance,
140        closest_approach_m=closest,
141        closest_approach_t=ts[closest_i] - ts[0],
142        overshoot_m=overshoot,
143        settling_time_s=settling_time,
144        settle_tolerance_m=tol,
145        within_radius_pct=within,
146        steady_mean_m=_mean(tail_d),
147        steady_rms_m=_rms(tail_d),
148        steady_max_m=max(tail_d) if tail_d else math.nan,
149        steady_peak_to_peak_m=(max(tail_d) - min(tail_d)) if tail_d else math.nan,
150        thrust_mean=_mean(thrust),
151        thrust_abs_mean=_mean([abs(x) for x in thrust]),
152        reverse_fraction=rev_frac,
153        thrust_reversals=reversals,
154        control_effort=effort,
155    )

def heading_metrics( log: SimLog, target_deg: float, *, start_t: float = 0.0, tail_seconds: float = 15.0, settle_tolerance_deg: float = 3.0) -> HeadingMetrics: 
174def heading_metrics(
175    log: SimLog,
176    target_deg: float,
177    *,
178    start_t: float = 0.0,
179    tail_seconds: float = 15.0,
180    settle_tolerance_deg: float = 3.0,
181) -> HeadingMetrics:
182    samples = [s for s in log.samples if s.t >= start_t]
183    if not samples:
184        raise ValueError("no samples after start_t")
185    ts = [s.t for s in samples]
186    err = [angle_difference(s.truth_heading, target_deg) for s in samples]  # signed
187    abserr = [abs(e) for e in err]
188    initial = abserr[0] if abserr else 0.0
189
190    rise = math.nan
191    for i, e in enumerate(abserr):
192        if e <= 0.1 * initial:
193            rise = ts[i] - ts[0]
194            break
195
196    # Overshoot: furthest the heading went *past* the target (opposite sign to
197    # the initial error).
198    init_sign = 1.0 if err[0] >= 0 else -1.0
199    past = [(-init_sign) * e for e in err]  # positive = beyond target
200    overshoot = max(0.0, max(past))
201
202    settling = math.nan
203    for i in range(len(abserr)):
204        if abserr[i] <= settle_tolerance_deg and all(
205            a <= settle_tolerance_deg for a in abserr[i:]
206        ):
207            settling = ts[i] - ts[0]
208            break
209
210    tail = [abs(angle_difference(s.truth_heading, target_deg)) for s in samples if s.t >= ts[-1] - tail_seconds]
211    return HeadingMetrics(
212        target_deg=target_deg,
213        rise_time_s=rise,
214        overshoot_deg=overshoot,
215        settling_time_s=settling,
216        settle_tolerance_deg=settle_tolerance_deg,
217        steady_error_deg=_mean(tail),
218    )
def steering_activity( log: SimLog, max_steer_angle_deg: float = 35.0) -> SteeringActivity: 
254def steering_activity(log: SimLog, max_steer_angle_deg: float = 35.0) -> SteeringActivity:
255    """Rotation rate / reversals of the steering command (``steering`` in [-1,1]
256    maps to +/-``max_steer_angle_deg`` of head rotation)."""
257    s = log.series("steering")
258    t = log.times()
259    rates: list[float] = []
260    reversals = 0
261    last_sign = 0
262    last, last_t = s[0], t[0]
263    for i in range(1, len(s)):
264        if s[i] != last:
265            rates.append(abs(s[i] - last) * max_steer_angle_deg / (t[i] - last_t))
266            sign = 1 if s[i] > last else -1
267            if last_sign and sign != last_sign:
268                reversals += 1
269            last_sign, last, last_t = sign, s[i], t[i]
270    dur = (t[-1] - t[0]) or 1.0
271    return SteeringActivity(
272        max_rate_dps=max(rates) if rates else 0.0,
273        mean_rate_dps=(sum(rates) / len(rates)) if rates else 0.0,
274        reversals_per_s=reversals / dur,
275    )

def sparkline(values: list[float], width: int = 50) -> str: 
24def sparkline(values: list[float], width: int = 50) -> str:
25    """A compact unicode sparkline of a series (NaNs render as spaces)."""
26    clean = [v for v in values if not math.isnan(v)]
27    if not clean:
28        return ""
29    lo, hi = min(clean), max(clean)
30    span = (hi - lo) or 1.0
31    # Down-sample to ``width`` buckets by averaging.
32    n = len(values)
33    out = []
34    for i in range(width):
35        a = i * n // width
36        b = max(a + 1, (i + 1) * n // width)
37        chunk = [v for v in values[a:b] if not math.isnan(v)]
38        if not chunk:
39            out.append(" ")
40            continue
41        avg = sum(chunk) / len(chunk)
42        idx = int((avg - lo) / span * (len(_BLOCKS) - 1))
43        out.append(_BLOCKS[min(len(_BLOCKS) - 1, max(0, idx))])
44    return "".join(out)

def optimize( cost_fn: Callable[[dict], float], params: list[Param], *, max_evals: int = 80, init_step_frac: float = 0.4, shrink: float = 0.5, min_step_frac: float = 0.01) -> tuple[dict, float, int]: 
46def optimize(
47    cost_fn: Callable[[dict], float],
48    params: list[Param],
49    *,
50    max_evals: int = 80,
51    init_step_frac: float = 0.4,
52    shrink: float = 0.5,
53    min_step_frac: float = 0.01,
54) -> tuple[dict, float, int]:
55    """Coordinate-descent / pattern search minimising ``cost_fn``.
56
57    For each parameter it probes +/- a step; on improvement it moves there, and
58    when a full sweep yields no improvement it shrinks every step. Deterministic
59    and dependency-free. Returns ``(best_point, best_cost, n_evals)``.
60    """
61    x = {p.name: p.init for p in params}
62    step = {p.name: (p.high - p.low) * init_step_frac for p in params}
63    cache: dict[tuple, float] = {}
64
65    def evaluate(point: dict) -> float:
66        key = tuple(round(point[p.name], 6) for p in params)
67        if key not in cache:
68            cache[key] = cost_fn(point)
69        return cache[key]
70
71    best = evaluate(x)
72    evals = 1
73    while evals < max_evals:
74        improved = False
75        for p in params:
76            for direction in (1.0, -1.0):
77                if evals >= max_evals:
78                    break
79                cand = dict(x)
80                cand[p.name] = min(p.high, max(p.low, x[p.name] + direction * step[p.name]))
81                if abs(cand[p.name] - x[p.name]) < 1e-12:
82                    continue
83                c = evaluate(cand)
84                evals += 1
85                if c < best:
86                    x, best = cand, c
87                    improved = True
88                    break
89        if not improved:
90            for p in params:
91                step[p.name] *= shrink
92            if all(step[p.name] <= (p.high - p.low) * min_step_frac for p in params):
93                break
94    return x, best, evals
