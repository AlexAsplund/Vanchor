 47    def __init__(self, min_distance_m: float = 3.0, max_points: int = 60000) -> None:
 48        self.min_distance_m = min_distance_m
 49        self.max_points = max_points

 51        self.points: list[tuple[float, float, float]] = []
 52        # Parallel bottom-hardness layer (lat, lon, hardness_index) from imported

 65    # -- persistence ------------------------------------------------------ #
 66    # Recorded/imported SOUNDINGS live in the small depthmap.json (rewritten
 67    # often by the recorder). The large STATIC imported chart (hardness /
 68    # contours / composition) lives in a SEPARATE file written only on import,
 69    # so the recorder's periodic save stays tiny and never rewrites the big
 70    # chart (which was both slow and corruption-prone mid-write).
 71    @staticmethod
 72    def _atomic_write(path: str, obj: dict) -> None:
 73        os.makedirs(os.path.dirname(path) or ".", exist_ok=True)
 74        tmp = path + ".tmp"
 75        with open(tmp, "wb") as fh:
 76            fh.write(_json_dumps(obj))
 77        os.replace(tmp, path)   # atomic: a kill/power-loss mid-write can't corrupt it

 80        """Persist the soundings (small; called often by the recorder)."""

100    def load(self, path: str, chart_path: str | None = None) -> None:
101        if os.path.exists(path):                      # soundings
102            try:
103                with open(path, "rb") as fh:
104                    obj = _json_loads(fh.read())
105                self.points = [tuple(p) for p in obj.get("points", []) if len(p) == 3][-self.max_points:]
106                if self.points:
107                    la, lo, _ = self.points[-1]
108                    self._last = GeoPoint(la, lo)
109            except (OSError, ValueError, TypeError) as exc:  # pragma: no cover
110                logger.warning("could not load depth soundings: %s", exc)
111        if chart_path and os.path.exists(chart_path):  # static chart (separate file)
112            try:
113                with open(chart_path, "rb") as fh:
114                    ch = _json_loads(fh.read())
115                self.hardness = [tuple(p) for p in ch.get("hardness", []) if len(p) == 3][-self.max_points:]
116                self.contours = ch.get("contours", []) or []
117                self.composition = ch.get("composition", []) or []
118            except (OSError, ValueError, TypeError) as exc:  # pragma: no cover
119                logger.warning("could not load depth chart: %s", exc)
120        logger.info("loaded %d soundings, %d hardness, %d contours, %d composition",
121                    len(self.points), len(self.hardness), len(self.contours), len(self.composition))

123    def record(self, point: GeoPoint | None, depth_m: float) -> None:
124        if point is None or depth_m <= 0.0:
125            return
126        if self._last is None or haversine_m(self._last, point) >= self.min_distance_m:
127            self.points.append((point.lat, point.lon, depth_m))
128            self._last = point
129            if len(self.points) > self.max_points:
130                self.points = self.points[-self.max_points :]

137        self,
138        bbox: tuple[float, float, float, float] | None = None,

141        """Imported depth contours, windowed to a (west, south, east, north)

181    def as_grid(
182        self,
183        cell_m: float = 15.0,
184        max_cells: int = 3000,
185        interpolate: bool = True,
186        interp_radius: int = 5,
187        interp_min_dirs: int = 6,
188        interp_power: float = 2.0,
189        radiate: bool = True,
190        radiate_radius_m: float = 30.0,
191        bbox: tuple[float, float, float, float] | None = None,
192        source: list[tuple[float, float, float]] | None = None,
193    ) -> dict:
194        """Bin every sounding into a square grid (~``cell_m`` metres) and average
195        depth per cell, returning a compact structure for the UI to colour-scale.
196
197        Soundings span a small area, so we bin in a local metric frame using a
198        flat metres-per-degree conversion at the data's mean latitude (cheap and
199        accurate over the breadcrumb's extent -- no pyproj needed). Binning is a
200        single O(n) pass over the points.
201
202        The returned cell count is capped at ``max_cells`` by growing the
203        effective cell size (doubling until the bins fit), and the cell size
204        actually used is reported back so the client can label its colour scale.
205
206        Two fill passes spread the measured data into the empty cells around it,
207        in order of confidence:
208
209        * **Radiate (nearest-neighbour / Voronoi).** When ``radiate`` is on
210          (default), each empty cell within ``radiate_radius_m`` metres of a
211          measured cell is assigned the depth of the *nearest* measured cell --
212          the bottom is assumed roughly constant out to the Voronoi boundary
213          where a different reading becomes nearer. The radius is bounded (a
214          few cells) so one ping can't paint a whole lake and can't bleed into a
215          neighbouring waterbody across an empty gap wider than the radius.
216          Radiated cells are confident assumptions: ``"kind": "radiated"`` and
217          ``"est": false``.
218
219        * **Interpolate (enclosed-gap IDW).** When ``interpolate`` is on
220          (default), *enclosed* empty cells -- holes surrounded by measured data
221          in at least ``interp_min_dirs`` of the 8 compass directions within
222          ``interp_radius`` cells -- get an inverse-distance-weighted estimate
223          blended from the *differing* readings around them. These are genuine
224          guesses between differing soundings: ``"kind": "interp"`` and
225          ``"est": true``. Interp takes priority over radiate on any cell that
226          qualifies, since a blend between differing readings is more honest
227          there than picking the single nearest one.
228
229        Measured cells carry ``"kind": "measured"`` and ``"est": false``. The
230        deep middle of a sparsely-edged lake and any separate, far-away cluster
231        stay untouched by both passes.
232
233        Returns ``{cell_m, min_depth, max_depth, count, cells}`` where each cell
234        is ``{"lat", "lon", "depth", "n", "est", "kind"}`` at the cell centre.
235        """
236        cell_m = max(1.0, float(cell_m))
237        # ``source`` selects which layer to grid (defaults to depth ``points``;
238        # pass ``self.hardness`` for the bottom-hardness grid -- same (lat, lon,
239        # value) shape, so the binning/fill/windowing below is identical).
240        # Tier-1 viewport windowing: with a bbox (west, south, east, north) only
241        # the soundings inside it are gridded, so the client fetches just what is
242        # on screen (+ its scroll padding) instead of the whole chart.
243        pts = self.points if source is None else source
244        if bbox is not None:
245            w, s, e, n = bbox
246            pts = [p for p in pts if s <= p[0] <= n and w <= p[1] <= e]
247        if not pts:
248            return {
249                "cell_m": cell_m,
250                "min_depth": 0.0,
251                "max_depth": 0.0,
252                "count": 0,
253                "cells": [],
254            }
255
256        lats = [la for la, _, _ in pts]
257        lons = [lo for _, lo, _ in pts]
258        mean_lat = sum(lats) / len(lats)
259        m_per_deg_lat = _M_PER_DEG_LAT
260        m_per_deg_lon = _M_PER_DEG_LAT * max(0.01, math.cos(math.radians(mean_lat)))
261
262        # Reference corner so cell indices are small non-negative integers.
263        lat0, lon0 = min(lats), min(lons)
264
265        # Grow the cell size until the binned cell count fits under the cap. The
266        # number of distinct cells only ever shrinks as the cell grows, so a few
267        # doublings converge quickly.
268        while True:
269            dlat = cell_m / m_per_deg_lat
270            dlon = cell_m / m_per_deg_lon
271            # accumulator: (i, j) -> [sum_depth, count]
272            acc: dict[tuple[int, int], list[float]] = {}
273            for la, lo, d in pts:
274                i = int((la - lat0) / dlat)
275                j = int((lo - lon0) / dlon)
276                bucket = acc.get((i, j))
277                if bucket is None:
278                    acc[(i, j)] = [d, 1.0]
279                else:
280                    bucket[0] += d
281                    bucket[1] += 1.0
282            if len(acc) <= max_cells:
283                break
284            cell_m *= 2.0
285
286        # Measured cells: (i, j) -> averaged depth.
287        measured: dict[tuple[int, int], float] = {
288            ij: sum_d / n for ij, (sum_d, n) in acc.items()
289        }
290
291        # -- interpolation pass: fill enclosed holes (uncertain guesses) ---- #
292        # Estimated cells: (i, j) -> idw depth. Built only for empty cells that
293        # are genuinely surrounded by *differing* measured data within
294        # ``interp_radius``. These are the honest "between two readings" guesses.
295        estimated: dict[tuple[int, int], float] = {}
296        if interpolate and len(measured) >= 8 and len(measured) < max_cells:
297            estimated = self._interpolate_holes(
298                measured,
299                radius=max(1, int(interp_radius)),
300                min_dirs=max(1, min(8, int(interp_min_dirs))),
301                power=float(interp_power),
302                budget=max_cells - len(measured),
303            )
304
305        # -- radiate pass: nearest-neighbour (Voronoi) footprint ------------ #
306        # Each empty cell within ``radiate_radius_m`` of a measured cell takes
307        # the depth of the NEAREST measured cell. Bounded radius => one ping
308        # paints only a few cells and can't cross an empty gap to a neighbouring
309        # waterbody. Interp cells already claimed above take priority and are
310        # skipped here. Remaining cell budget is shared after interp.
311        radiated: dict[tuple[int, int], float] = {}
312        radiate_cells = int(radiate_radius_m / cell_m)  # radius in cells
313        budget = max_cells - len(measured) - len(estimated)
314        if radiate and radiate_cells >= 1 and measured and budget > 0:
315            radiated = self._radiate(
316                measured,
317                claimed=estimated,
318                radius=radiate_cells,
319                budget=budget,
320            )
321
322        cells: list[dict] = []
323        min_depth = math.inf
324        max_depth = -math.inf
325        for (i, j), avg in measured.items():
326            min_depth = min(min_depth, avg)
327            max_depth = max(max_depth, avg)
328            cells.append(
329                {
330                    "lat": lat0 + (i + 0.5) * dlat,
331                    "lon": lon0 + (j + 0.5) * dlon,
332                    "depth": avg,
333                    "n": int(acc[(i, j)][1]),
334                    "est": False,
335                    "kind": "measured",
336                }
337            )
338        for (i, j), avg in radiated.items():
339            min_depth = min(min_depth, avg)
340            max_depth = max(max_depth, avg)
341            cells.append(
342                {
343                    "lat": lat0 + (i + 0.5) * dlat,
344                    "lon": lon0 + (j + 0.5) * dlon,
345                    "depth": avg,
346                    "n": 0,
347                    "est": False,
348                    "kind": "radiated",
349                }
350            )
351        for (i, j), avg in estimated.items():
352            min_depth = min(min_depth, avg)
353            max_depth = max(max_depth, avg)
354            cells.append(
355                {
356                    "lat": lat0 + (i + 0.5) * dlat,
357                    "lon": lon0 + (j + 0.5) * dlon,
358                    "depth": avg,
359                    "n": 0,
360                    "est": True,
361                    "kind": "interp",
362                }
363            )
364
365        return {
366            "cell_m": cell_m,
367            "min_depth": min_depth,
368            "max_depth": max_depth,
369            "count": len(self.points),
370            "cells": cells,
371        }

373    @staticmethod
374    def _interpolate_holes(
375        measured: dict[tuple[int, int], float],
376        radius: int,
377        min_dirs: int,
378        power: float,
379        budget: int,
380    ) -> dict[tuple[int, int], float]:
381        """Fill empty cells that are enclosed by measured cells via IDW.
382
383        A cell counts as *enclosed* only when, scanning outward along the 8
384        compass directions up to ``radius`` cells, a measured cell is met in at
385        least ``min_dirs`` of those 8 directions. This is the guard that keeps
386        us inside the data: the deep middle of a lake whose only soundings hug
387        the shore fails the test (no measured cell within ``radius`` straight
388        out), and an isolated cluster never reaches another cluster's cells.
389
390        ``budget`` caps how many estimates we add so the total cell count stays
391        under ``max_cells``; once exhausted we stop filling.
392        """
393        if budget <= 0:
394            return {}
395
396        # 8 compass directions as (di, dj).
397        dirs = (
398            (1, 0), (-1, 0), (0, 1), (0, -1),
399            (1, 1), (1, -1), (-1, 1), (-1, -1),
400        )
401
402        # Candidate empty cells: only those within ``radius`` of a measured cell
403        # (an enclosed cell must have measured neighbours nearby), collected from
404        # the measured cells' neighbourhoods. This bounds the work at
405        # O(measured * radius^2) -- INDEPENDENT of how widely the soundings are
406        # spread. (Scanning the full measured bounding box is O(bbox_area *
407        # radius^2), which explodes for sparse wide data -- a few thousand
408        # soundings over a whole lake -> tens of seconds, freezing the single-
409        # threaded event loop on one /api/depth/grid request.)
410        candidates: set[tuple[int, int]] = set()
411        for ci, cj in measured:
412            for di in range(-radius, radius + 1):
413                for dj in range(-radius, radius + 1):
414                    cell = (ci + di, cj + dj)
415                    if cell not in measured:
416                        candidates.add(cell)
417
418        estimated: dict[tuple[int, int], float] = {}
419        for i, j in candidates:
420            # Enclosure test: count directions with a measured hit in range.
421            hits = 0
422            for di, dj in dirs:
423                for r in range(1, radius + 1):
424                    if (i + di * r, j + dj * r) in measured:
425                        hits += 1
426                        break
427                if hits >= min_dirs:
428                    break  # already enclosed; no need to scan the rest
429            if hits < min_dirs:
430                continue
431
432            # IDW over measured cells within the (square) radius window.
433            wsum = 0.0
434            dsum = 0.0
435            for di in range(-radius, radius + 1):
436                for dj in range(-radius, radius + 1):
437                    if di == 0 and dj == 0:
438                        continue
439                    d = measured.get((i + di, j + dj))
440                    if d is None:
441                        continue
442                    dist2 = di * di + dj * dj
443                    w = 1.0 / (dist2 ** (power / 2.0))
444                    wsum += w
445                    dsum += w * d
446            if wsum > 0.0:
447                estimated[(i, j)] = dsum / wsum
448                if len(estimated) >= budget:
449                    return estimated
450        return estimated

452    @staticmethod
453    def _radiate(
454        measured: dict[tuple[int, int], float],
455        claimed: dict[tuple[int, int], float],
456        radius: int,
457        budget: int,
458    ) -> dict[tuple[int, int], float]:
459        """Fill empty cells with the depth of the NEAREST measured cell.
460
461        This is a bounded nearest-neighbour (Voronoi) extension: every empty
462        cell within ``radius`` cells of at least one measured cell is assigned
463        the depth of the closest measured cell (Chebyshev-windowed, Euclidean
464        tie-break). The bottom is assumed roughly the same as the nearest
465        sounding out to the Voronoi boundary -- the locus where a *different*
466        reading becomes nearer.
467
468        The radius is the critical guard: a cell more than ``radius`` cells from
469        every sounding is never filled, so a single ping paints only its own
470        small neighbourhood (it can't flood a whole lake) and the fill can't
471        jump an empty gap wider than ``radius`` into a neighbouring waterbody.
472
473        ``claimed`` cells (already taken by the enclosed-gap interpolation) are
474        skipped so interp's honest blend wins where readings differ. ``budget``
475        caps how many cells we add to respect ``max_cells``.
476        """
477        if budget <= 0:
478            return {}
479
480        # Iterate the MEASURED cells and paint their square ``radius`` window of
481        # empty cells, keeping the NEAREST measured cell per empty cell. This is
482        # O(measured * radius^2) -- bounded by ``max_cells`` -- and INDEPENDENT of
483        # how widely or sparsely the soundings are spread. (Scanning the measured
484        # bounding box instead is O(bbox_area * radius^2), which explodes for
485        # sparse wide data -- e.g. a few thousand soundings over a whole lake at a
486        # small cell size -- and stalls the single-threaded event loop for
487        # seconds per /api/depth/grid request.)
488        best: dict[tuple[int, int], tuple[int, float]] = {}  # cell -> (dist2, depth)
489        for (i, j), depth in measured.items():
490            for di in range(-radius, radius + 1):
491                for dj in range(-radius, radius + 1):
492                    if di == 0 and dj == 0:
493                        continue
494                    cell = (i + di, j + dj)
495                    if cell in measured or cell in claimed:
496                        continue
497                    d2 = di * di + dj * dj
498                    cur = best.get(cell)
499                    if cur is None or d2 < cur[0]:
500                        best[cell] = (d2, depth)
501
502        radiated: dict[tuple[int, int], float] = {}
503        for cell, (_, depth) in best.items():
504            radiated[cell] = depth
505            if len(radiated) >= budget:
506                break
507        return radiated

374    @staticmethod
375    def _interpolate_holes(
376        measured: dict[tuple[int, int], float],
377        radius: int,
378        min_dirs: int,
379        power: float,
380        budget: int,
381    ) -> dict[tuple[int, int], float]:
382        """Fill empty cells that are enclosed by measured cells via IDW.
383
384        A cell counts as *enclosed* only when, scanning outward along the 8
385        compass directions up to ``radius`` cells, a measured cell is met in at
386        least ``min_dirs`` of those 8 directions. This is the guard that keeps
387        us inside the data: the deep middle of a lake whose only soundings hug
388        the shore fails the test (no measured cell within ``radius`` straight
389        out), and an isolated cluster never reaches another cluster's cells.
390
391        ``budget`` caps how many estimates we add so the total cell count stays
392        under ``max_cells``; once exhausted we stop filling.
393        """
394        if budget <= 0:
395            return {}
396
397        # 8 compass directions as (di, dj).
398        dirs = (
399            (1, 0), (-1, 0), (0, 1), (0, -1),
400            (1, 1), (1, -1), (-1, 1), (-1, -1),
401        )
402
403        # Candidate empty cells: only those within ``radius`` of a measured cell
404        # (an enclosed cell must have measured neighbours nearby), collected from
405        # the measured cells' neighbourhoods. This bounds the work at
406        # O(measured * radius^2) -- INDEPENDENT of how widely the soundings are
407        # spread. (Scanning the full measured bounding box is O(bbox_area *
408        # radius^2), which explodes for sparse wide data -- a few thousand
409        # soundings over a whole lake -> tens of seconds, freezing the single-
410        # threaded event loop on one /api/depth/grid request.)
411        candidates: set[tuple[int, int]] = set()
412        for ci, cj in measured:
413            for di in range(-radius, radius + 1):
414                for dj in range(-radius, radius + 1):
415                    cell = (ci + di, cj + dj)
416                    if cell not in measured:
417                        candidates.add(cell)
418
419        estimated: dict[tuple[int, int], float] = {}
420        for i, j in candidates:
421            # Enclosure test: count directions with a measured hit in range.
422            hits = 0
423            for di, dj in dirs:
424                for r in range(1, radius + 1):
425                    if (i + di * r, j + dj * r) in measured:
426                        hits += 1
427                        break
428                if hits >= min_dirs:
429                    break  # already enclosed; no need to scan the rest
430            if hits < min_dirs:
431                continue
432
433            # IDW over measured cells within the (square) radius window.
434            wsum = 0.0
435            dsum = 0.0
436            for di in range(-radius, radius + 1):
437                for dj in range(-radius, radius + 1):
438                    if di == 0 and dj == 0:
439                        continue
440                    d = measured.get((i + di, j + dj))
441                    if d is None:
442                        continue
443                    dist2 = di * di + dj * dj
444                    w = 1.0 / (dist2 ** (power / 2.0))
445                    wsum += w
446                    dsum += w * d
447            if wsum > 0.0:
448                estimated[(i, j)] = dsum / wsum
449                if len(estimated) >= budget:
450                    return estimated
451        return estimated

465        tie-break). The bottom is assumed roughly the same as the nearest
466        sounding out to the Voronoi boundary -- the locus where a *different*
467        reading becomes nearer.
468
469        The radius is the critical guard: a cell more than ``radius`` cells from
470        every sounding is never filled, so a single ping paints only its own
471        small neighbourhood (it can't flood a whole lake) and the fill can't
472        jump an empty gap wider than ``radius`` into a neighbouring waterbody.
473
474        ``claimed`` cells (already taken by the enclosed-gap interpolation) are
475        skipped so interp's honest blend wins where readings differ. ``budget``
476        caps how many cells we add to respect ``max_cells``.
477        """
478        if budget <= 0:
479            return {}
480
481        # Iterate the MEASURED cells and paint their square ``radius`` window of
482        # empty cells, keeping the NEAREST measured cell per empty cell. This is
483        # O(measured * radius^2) -- bounded by ``max_cells`` -- and INDEPENDENT of
484        # how widely or sparsely the soundings are spread. (Scanning the measured
485        # bounding box instead is O(bbox_area * radius^2), which explodes for
486        # sparse wide data -- e.g. a few thousand soundings over a whole lake at a
487        # small cell size -- and stalls the single-threaded event loop for
488        # seconds per /api/depth/grid request.)
489        best: dict[tuple[int, int], tuple[int, float]] = {}  # cell -> (dist2, depth)
490        for (i, j), depth in measured.items():
491            for di in range(-radius, radius + 1):
492                for dj in range(-radius, radius + 1):
493                    if di == 0 and dj == 0:
494                        continue
495                    cell = (i + di, j + dj)
496                    if cell in measured or cell in claimed:
497                        continue
498                    d2 = di * di + dj * dj
499                    cur = best.get(cell)
500                    if cur is None or d2 < cur[0]:
501                        best[cell] = (d2, depth)
502
503        radiated: dict[tuple[int, int], float] = {}
504        for cell, (_, depth) in best.items():
505            radiated[cell] = depth
506            if len(radiated) >= budget:
507                break
508        return radiated

195        """Bin every sounding into a square grid (~``cell_m`` metres) and average
196        depth per cell, returning a compact structure for the UI to colour-scale.
197
198        Soundings span a small area, so we bin in a local metric frame using a
199        flat metres-per-degree conversion at the data's mean latitude (cheap and
200        accurate over the breadcrumb's extent -- no pyproj needed). Binning is a
201        single O(n) pass over the points.
202
203        The returned cell count is capped at ``max_cells`` by growing the
204        effective cell size (doubling until the bins fit), and the cell size
205        actually used is reported back so the client can label its colour scale.
206
207        Two fill passes spread the measured data into the empty cells around it,
208        in order of confidence:
209
210        * **Radiate (nearest-neighbour / Voronoi).** When ``radiate`` is on
211          (default), each empty cell within ``radiate_radius_m`` metres of a
212          measured cell is assigned the depth of the *nearest* measured cell --

214          where a different reading becomes nearer. The radius is bounded (a
215          few cells) so one ping can't paint a whole lake and can't bleed into a
216          neighbouring waterbody across an empty gap wider than the radius.

218          ``"est": false``.

221          (default), *enclosed* empty cells -- holes surrounded by measured data
222          in at least ``interp_min_dirs`` of the 8 compass directions within

224          blended from the *differing* readings around them. These are genuine
225          guesses between differing soundings: ``"kind": "interp"`` and

227          qualifies, since a blend between differing readings is more honest
228          there than picking the single nearest one.

231        deep middle of a sparsely-edged lake and any separate, far-away cluster
232        stay untouched by both passes.
233

235        is ``{"lat", "lon", "depth", "n", "est", "kind"}`` at the cell centre.
236        """
237        cell_m = max(1.0, float(cell_m))
238        # ``source`` selects which layer to grid (defaults to depth ``points``;
239        # pass ``self.hardness`` for the bottom-hardness grid -- same (lat, lon,
240        # value) shape, so the binning/fill/windowing below is identical).
241        # Tier-1 viewport windowing: with a bbox (west, south, east, north) only

336                    "kind": "measured",
337                }

341            max_depth = max(max_depth, avg)
342            cells.append(
343                {
344                    "lat": lat0 + (i + 0.5) * dlat,
345                    "lon": lon0 + (j + 0.5) * dlon,
346                    "depth": avg,
347                    "n": 0,
348                    "est": False,
349                    "kind": "radiated",
350                }
351            )
352        for (i, j), avg in estimated.items():
353            min_depth = min(min_depth, avg)
354            max_depth = max(max_depth, avg)
355            cells.append(
356                {
357                    "lat": lat0 + (i + 0.5) * dlat,
358                    "lon": lon0 + (j + 0.5) * dlon,
