vanchor.nav.trip

Trip log: record each outing's track, distance, duration and speed stats.

A trip is one continuous outing -- from the moment the boat starts making way until it goes idle (or the helmsman stops it). While a trip is active the TripLog samples the boat's position into a breadcrumb track (min-distance filtered like ~vanchor.nav.track.TrackRecorder), integrates the distance travelled as the sum of segment lengths, and tracks the max speed-over-ground.

The log can auto-start a trip when the boat first makes way (SOG over a small threshold) and auto-stop it after a stretch of idleness; a manual start/stop always works and overrides the automatic behaviour. Finished trips are persisted to <data_dir>/trips/<id>.json and can be listed, fetched, exported as GPX or deleted.

All time comes in through now arguments (the Runtime feeds its injectable _now_fn), so the auto-start/stop logic is fully deterministic in tests.

 38@dataclass
 39class Trip:
 40    """One outing's recorded track and summary statistics."""
 41
 42    id: str
 43    name: str
 44    started_at: float
 45    ended_at: float | None = None
 46    distance_m: float = 0.0
 47    max_speed_kn: float = 0.0
 48    points: list[GeoPoint] = field(default_factory=list)
 49    # Whether this trip was started automatically (so auto-stop may end it).
 50    auto: bool = False
 51
 52    def duration_s(self, now: float) -> float:
 53        end = self.ended_at if self.ended_at is not None else now
 54        return max(0.0, end - self.started_at)
 55
 56    def avg_speed_kn(self, now: float) -> float:
 57        dur = self.duration_s(now)
 58        if dur <= 0.0:
 59            return 0.0
 60        # metres-per-second -> knots
 61        return (self.distance_m / dur) / 0.514444
 62
 63    def summary(self, now: float) -> dict:
 64        """Summary fields only (no point array) for the list endpoint."""
 65        return {
 66            "id": self.id,
 67            "name": self.name,
 68            "started_at": self.started_at,
 69            "ended_at": self.ended_at,
 70            "distance_m": round(self.distance_m, 1),
 71            "duration_s": round(self.duration_s(now), 1),
 72            "avg_speed_kn": round(self.avg_speed_kn(now), 2),
 73            "max_speed_kn": round(self.max_speed_kn, 2),
 74            "point_count": len(self.points),
 75        }
 76
 77    def to_dict(self, now: float) -> dict:
 78        """Full record including the track points."""
 79        d = self.summary(now)
 80        d["points"] = [[p.lat, p.lon] for p in self.points]
 81        return d

 84class TripLog:
 85    """Records the current outing and persists finished trips to disk.
 86
 87    Call :meth:`update` on every telemetry tick with the boat's current position
 88    and speed-over-ground (knots) plus the current time; it handles breadcrumb
 89    sampling, distance/max-speed accumulation and the auto-start/stop state
 90    machine. :meth:`start`/:meth:`stop` give the helmsman manual control.
 91    """
 92
 93    def __init__(
 94        self,
 95        data_dir: str,
 96        *,
 97        min_distance_m: float = 5.0,
 98        auto: bool = True,
 99        start_speed_kn: float = 0.5,
100        idle_timeout_s: float = 120.0,
101        max_points: int = 5000,
102    ) -> None:
103        self.dir = os.path.join(data_dir, "trips")
104        self.min_distance_m = min_distance_m
105        self.auto = auto
106        self.start_speed_kn = start_speed_kn
107        self.idle_timeout_s = idle_timeout_s
108        self.max_points = max_points
109        self.current: Trip | None = None
110        # Wall-clock time the boat was last seen making way (for auto-stop).
111        self._last_moving_at: float | None = None
112
113    # ------------------------------------------------------------------ #
114    # Manual control
115    # ------------------------------------------------------------------ #
116    def start(self, name: str | None, now: float, *, auto: bool = False) -> Trip:
117        """Begin a fresh trip, finalizing any trip already in progress."""
118        if self.current is not None:
119            self.stop(now)
120        started = now
121        trip = Trip(
122            id=_trip_id(started),
123            name=name or time.strftime("Trip %Y-%m-%d %H:%M", time.localtime(started)),
124            started_at=started,
125            auto=auto,
126        )
127        self.current = trip
128        self._last_moving_at = now
129        logger.info("trip started: %s (%s)", trip.id, "auto" if auto else "manual")
130        return trip
131
132    def stop(self, now: float) -> Trip | None:
133        """Finalize the active trip and persist it. Returns the saved trip."""
134        trip = self.current
135        if trip is None:
136            return None
137        trip.ended_at = now
138        self.current = None
139        self._last_moving_at = None
140        self._save(trip, now)
141        logger.info(
142            "trip stopped: %s (%.0f m, %.0f s)",
143            trip.id,
144            trip.distance_m,
145            trip.duration_s(now),
146        )
147        return trip
148
149    # ------------------------------------------------------------------ #
150    # Per-tick recording + auto start/stop
151    # ------------------------------------------------------------------ #
152    def update(self, position: GeoPoint | None, sog_kn: float, now: float) -> None:
153        """Advance the trip log one tick.
154
155        Records a breadcrumb + integrates distance/max-speed for the active trip,
156        and runs the auto-start/stop machine when ``auto`` is enabled.
157        """
158        making_way = sog_kn >= self.start_speed_kn and position is not None
159        if making_way:
160            self._last_moving_at = now
161
162        if self.current is None:
163            # Auto-start when the boat first makes way.
164            if self.auto and making_way:
165                self.start(None, now, auto=True)
166            else:
167                return
168        else:
169            self._accumulate(position, sog_kn)
170            # Auto-stop an auto-started trip after a stretch of idleness.
171            if (
172                self.auto
173                and self.current.auto
174                and self._last_moving_at is not None
175                and now - self._last_moving_at >= self.idle_timeout_s
176            ):
177                self.stop(now)
178
179    def _accumulate(self, position: GeoPoint | None, sog_kn: float) -> None:
180        trip = self.current
181        assert trip is not None
182        if sog_kn > trip.max_speed_kn:
183            trip.max_speed_kn = sog_kn
184        if position is None:
185            return
186        if not trip.points:
187            trip.points.append(position)
188            return
189        seg = haversine_m(trip.points[-1], position)
190        if seg >= self.min_distance_m:
191            trip.distance_m += seg
192            trip.points.append(position)
193            if len(trip.points) > self.max_points:
194                trip.points = trip.points[-self.max_points :]
195
196    # ------------------------------------------------------------------ #
197    # Telemetry
198    # ------------------------------------------------------------------ #
199    def snapshot(self, now: float) -> dict:
200        """The CURRENT trip's live stats for telemetry (zeros when idle)."""
201        trip = self.current
202        if trip is None:
203            return {
204                "active": False,
205                "name": None,
206                "distance_m": 0.0,
207                "duration_s": 0.0,
208                "avg_speed_kn": 0.0,
209                "max_speed_kn": 0.0,
210            }
211        return {
212            "active": True,
213            "name": trip.name,
214            "distance_m": round(trip.distance_m, 1),
215            "duration_s": round(trip.duration_s(now), 1),
216            "avg_speed_kn": round(trip.avg_speed_kn(now), 2),
217            "max_speed_kn": round(trip.max_speed_kn, 2),
218        }
219
220    # ------------------------------------------------------------------ #
221    # Persistence + queries
222    # ------------------------------------------------------------------ #
223    def _path(self, trip_id: str) -> str:
224        return os.path.join(self.dir, f"{trip_id}.json")
225
226    def _save(self, trip: Trip, now: float) -> None:
227        try:
228            os.makedirs(self.dir, exist_ok=True)
229            with open(self._path(trip.id), "w", encoding="utf-8") as fh:
230                json.dump(trip.to_dict(now), fh)
231        except OSError as exc:  # pragma: no cover - disk failure
232            logger.warning("could not save trip %s: %s", trip.id, exc)
233
234    def list_trips(self) -> list[dict]:
235        """Summaries of all saved trips, newest first."""
236        out: list[dict] = []
237        if not os.path.isdir(self.dir):
238            return out
239        for name in sorted(os.listdir(self.dir)):
240            if not name.endswith(".json"):
241                continue
242            try:
243                with open(os.path.join(self.dir, name), encoding="utf-8") as fh:
244                    d = json.load(fh)
245            except (OSError, ValueError):
246                continue
247            d.pop("points", None)
248            out.append(d)
249        out.sort(key=lambda d: d.get("started_at", 0.0), reverse=True)
250        return out
251
252    def get_trip(self, trip_id: str) -> dict | None:
253        """Full saved trip (including points), or None if absent."""
254        path = self._path(trip_id)
255        if not os.path.isfile(path):
256            return None
257        try:
258            with open(path, encoding="utf-8") as fh:
259                return json.load(fh)
260        except (OSError, ValueError):
261            return None
262
263    def delete_trip(self, trip_id: str) -> bool:
264        """Remove a saved trip. Returns True if it existed."""
265        path = self._path(trip_id)
266        if not os.path.isfile(path):
267            return False
268        try:
269            os.remove(path)
270            return True
271        except OSError:  # pragma: no cover - disk failure
272            return False
273
274    def gpx(self, trip_id: str) -> str | None:
275        """Export a saved trip as a GPX ``<trk>``, or None if absent."""
276        trip = self.get_trip(trip_id)
277        if trip is None:
278            return None
279        return trip_to_gpx(trip)

282def trip_to_gpx(trip: dict) -> str:
283    """Render a trip dict (as persisted) into a GPX 1.1 document string."""
284    from xml.sax.saxutils import escape
285
286    pts = trip.get("points", [])
287    seg = "".join(
288        f'<trkpt lat="{lat}" lon="{lon}"></trkpt>' for lat, lon in pts
289    )
290    name = escape(str(trip.get("name", trip.get("id", "Trip"))))
291    return (
292        '<?xml version="1.0" encoding="UTF-8"?>'
293        '<gpx version="1.1" creator="vanchor-ng" '
294        'xmlns="http://www.topografix.com/GPX/1/1">'
295        f"<trk><name>{name}</name><trkseg>{seg}</trkseg></trk>"
296        "</gpx>"
297    )

154        """Advance the trip log one tick.
155
156        Records a breadcrumb + integrates distance/max-speed for the active trip,
157        and runs the auto-start/stop machine when ``auto`` is enabled.
158        """
159        making_way = sog_kn >= self.start_speed_kn and position is not None
160        if making_way:

162
163        if self.current is None:

165            if self.auto and making_way:

167            else:
168                return
169        else:

171            # Auto-stop an auto-started trip after a stretch of idleness.
172            if (
173                self.auto
174                and self.current.auto
175                and self._last_moving_at is not None
176                and now - self._last_moving_at >= self.idle_timeout_s
