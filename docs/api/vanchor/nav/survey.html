vanchor.nav.survey

Area survey route planning -- "map mode" boustrophedon coverage (#47).

Given a closed area polygon (lat/lon), compute a back-and-forth lawnmower coverage route that surveys the whole area with a settable spacing between parallel passes. The classic agricultural / robotics pattern for this is the boustrophedon decomposition: sweep a set of parallel lines across the area, clip each line to the polygon, then walk them in alternating direction so the end of one pass connects to the start of the next (a continuous "ox-plough" path -- Greek boustrophedon, "as the ox turns").

All geometry maths happens in a metric UTM CRS (see vanchor.nav.water); only the final waypoints are converted back to lat/lon.

The default sweep direction is the polygon's longest axis (so passes run the long way and there are fewer expensive turns), unless an explicit angle_deg is given. The number of passes is capped so a tiny spacing on a big area can't produce thousands of waypoints; when the cap is hit a clear message is returned.

 33# Cap the number of generated waypoints. A coverage route with more legs than
 34# this is almost always a mis-set (too-small) spacing; we refuse rather than
 35# return an unusably huge route the UI then has to load and the boat to follow.
 36# Warn (do NOT reject) above this many waypoints -- mapping a big area at tight
 37# spacing legitimately produces a lot of passes. A much higher absolute guard
 38# only stops a pathological tiny-spacing run from making an unusable route.

 40MAX_WAYPOINTS = 5000
 41# Minimum spacing (m); guards against zero/negative/absurd spacing producing an

 56def _longest_axis_angle(poly: Polygon) -> float:
 57    """Bearing (deg, math convention) of the polygon's longest edge of its
 58    minimum rotated rectangle -- a good default sweep direction.
 59
 60    Sweeping *along* the longer side means each pass is as long as possible and
 61    there are as few turns as possible.
 62    """
 63    rect = poly.minimum_rotated_rectangle
 64    if rect.geom_type != "Polygon":
 65        return 0.0
 66    coords = list(rect.exterior.coords)
 67    # The min rotated rectangle has 5 coords (closed); inspect its two adjacent
 68    # edge vectors and take the longer one's direction.
 69    best_len = -1.0
 70    best_ang = 0.0
 71    for i in range(len(coords) - 1):
 72        (x0, y0), (x1, y1) = coords[i], coords[i + 1]
 73        dx, dy = x1 - x0, y1 - y0
 74        length = math.hypot(dx, dy)
 75        if length > best_len:
 76            best_len = length
 77            best_ang = math.degrees(math.atan2(dy, dx))
 78    return best_ang

 81def _ordered_lines_for_geom(geom: BaseGeometry) -> list[LineString]:
 82    """Flatten a clip result (LineString / MultiLineString / empty) to a list of
 83    LineStrings, dropping degenerate (zero-length) pieces."""
 84    if geom.is_empty:
 85        return []
 86    if geom.geom_type == "LineString":
 87        return [geom] if geom.length > 0 else []
 88    if geom.geom_type == "MultiLineString":
 89        return [g for g in geom.geoms if g.length > 0]
 90    if geom.geom_type == "GeometryCollection":
 91        out: list[LineString] = []
 92        for g in geom.geoms:
 93            out.extend(_ordered_lines_for_geom(g))
 94        return out
 95    return []

 98def _sweep_lines(poly_m: Polygon, spacing_m: float, angle_deg: float) -> list[LineString]:
 99    """Parallel lines ``spacing_m`` apart at ``angle_deg`` clipped to the polygon.
100
101    Generated in a rotated frame (so the sweep direction is axis-aligned), where
102    parallel lines are simply horizontal lines at increasing y; each is clipped
103    to the polygon and the clipped chords ordered along the sweep so the route
104    reads bottom-to-top.
105    """
106    import shapely.affinity as aff
107
108    # Rotate the polygon so the desired sweep direction becomes the x-axis; then
109    # parallel passes are horizontal lines and clipping is trivial. Rotate the
110    # results back at the end.
111    rotated = aff.rotate(poly_m, -angle_deg, origin="centroid", use_radians=False)
112    minx, miny, maxx, maxy = rotated.bounds
113    # Each pass "covers" a strip +/- spacing/2 around it, so we inset the first
114    # and last pass half a spacing from the edges and centre the stack. The pass
115    # count is ceil(height/spacing): a 50 m band at 10 m spacing -> 5 passes at
116    # 5,15,25,35,45 m (each covering its 10 m strip). This is the standard
117    # lawnmower convention and is robust to sub-metre jitter in the bounds.
118    height = maxy - miny
119    n_passes = max(1, int(math.ceil(height / spacing_m - 1e-9)))
120    used = (n_passes - 1) * spacing_m
121    y0 = miny + (height - used) / 2.0
122    pad = (maxx - minx) * 0.05 + 1.0  # extend lines past the bbox before clipping
123
124    lines: list[LineString] = []
125    for i in range(n_passes):
126        y = y0 + i * spacing_m
127        scan = LineString([(minx - pad, y), (maxx + pad, y)])
128        clipped = rotated.intersection(scan)
129        for seg in _ordered_lines_for_geom(clipped):
130            # Order each chord left-to-right; boustrophedon flipping happens later.
131            xs = sorted(seg.coords, key=lambda c: c[0])
132            lines.append(LineString([xs[0], xs[-1]]))
133
134    # Rotate the chords back into the metric frame.
135    return [aff.rotate(ln, angle_deg, origin=poly_m.centroid, use_radians=False) for ln in lines]

138def plan_survey(
139    polygon_latlon: list,
140    spacing_m: float,
141    angle_deg: float | None = None,
142) -> SurveyResult:
143    """Plan a boustrophedon coverage route over a closed area polygon.
144
145    Parameters
146    ----------
147    polygon_latlon:
148        The survey area as a list of ``[lat, lon]`` vertices (a closed ring; the
149        ring need not repeat its first point).
150    spacing_m:
151        Distance between parallel passes, in metres.
152    angle_deg:
153        Sweep direction in **compass-ish math degrees in the metric frame**; if
154        omitted the polygon's longest axis is used (fewest, longest passes).
155
156    Returns ordered waypoints ``[{name, lat, lon}]`` (``WP1``.. / ``DEST``).
157    Pure CPU work (shapely); run it in an executor.
158    """
159    pts = [(float(lat), float(lon)) for lat, lon in polygon_latlon]
160    if len(pts) < 3:
161        return SurveyResult(False, message="A survey area needs at least 3 points.")
162    if spacing_m is None or spacing_m < MIN_SPACING_M:
163        return SurveyResult(
164            False, message=f"Spacing must be at least {MIN_SPACING_M:g} m."
165        )
166
167    # Project to a metric frame centred on the polygon (use the first vertex).
168    proj = Projection.for_point(pts[0][1], pts[0][0])
169    ring_m = [proj.point_to_metric(lon, lat) for lat, lon in pts]
170    poly_m = Polygon(ring_m)
171    if not poly_m.is_valid:
172        poly_m = poly_m.buffer(0)
173    if poly_m.is_empty or poly_m.geom_type != "Polygon":
174        # buffer(0) of a self-intersecting ring may yield a MultiPolygon; take
175        # the largest part.
176        if poly_m.geom_type == "MultiPolygon" and not poly_m.is_empty:
177            poly_m = max(poly_m.geoms, key=lambda g: g.area)
178        else:
179            return SurveyResult(False, message="Survey area polygon is degenerate.")
180
181    sweep_ang = _longest_axis_angle(poly_m) if angle_deg is None else float(angle_deg)
182
183    lines = _sweep_lines(poly_m, float(spacing_m), sweep_ang)
184    if not lines:
185        return SurveyResult(
186            False,
187            message="Spacing too large for this area -- no passes fit.",
188            spacing_m=float(spacing_m),
189            angle_deg=sweep_ang,
190        )
191
192    # Boustrophedon ordering: walk the passes in sweep order, flipping every
193    # other pass so the end of one connects to the start of the next.
194    ordered_pts: list[tuple[float, float]] = []
195    prev_end: tuple[float, float] | None = None
196    for ln in lines:
197        a, b = ln.coords[0], ln.coords[-1]
198        if prev_end is not None:
199            # Connect to whichever end of this pass is nearer the previous end,
200            # so the connecting leg is the short turn (true boustrophedon).
201            d_a = math.hypot(a[0] - prev_end[0], a[1] - prev_end[1])
202            d_b = math.hypot(b[0] - prev_end[0], b[1] - prev_end[1])
203            if d_b < d_a:
204                a, b = b, a
205        ordered_pts.append(a)
206        ordered_pts.append(b)
207        prev_end = b
208
209    if len(ordered_pts) > MAX_WAYPOINTS:
210        return SurveyResult(
211            False,
212            message=(
213                f"{len(ordered_pts)} waypoints is too many to plan ({len(lines)} passes); "
214                f"increase the spacing."
215            ),
216            passes=len(lines),
217            spacing_m=float(spacing_m),
218            angle_deg=sweep_ang,
219        )
220
221    waypoints: list[dict] = []
222    n = len(ordered_pts)
223    for i, (x, y) in enumerate(ordered_pts):
224        lon, lat = proj.point_to_lonlat(x, y)
225        name = "DEST" if i == n - 1 else f"WP{i + 1}"
226        waypoints.append({"name": name, "lat": lat, "lon": lon})
227
228    warn = (
229        f" Heads up: {len(waypoints)} waypoints is a lot to run -- wider spacing is easier."
230        if len(waypoints) > WARN_WAYPOINTS else ""
231    )
232    return SurveyResult(
233        ok=True,
234        waypoints=waypoints,
235        message=(
236            f"Survey route: {len(lines)} passes, {len(waypoints)} waypoints, "
237            f"{spacing_m:g} m spacing.{warn}"
238        ),
239        passes=len(lines),
240        spacing_m=float(spacing_m),
241        angle_deg=sweep_ang,
242    )

252def plan_work_spots(polygon_latlon: list, spacing_m: float) -> SurveyResult:
253    """Even grid of work spots inside an area polygon (``[[lat, lon], ...]`` ring),
254    spaced ~``spacing_m`` apart and ordered serpentine (lawnmower) so the boat
255    works them in a tidy sweep. Returns ``SurveyResult`` with waypoints
256    ``[{name, lat, lon}]``. Pure CPU (shapely); run in an executor. Water-clipping
257    is applied by the caller (``Runtime.plan_work_spots``)."""
258    pts = [(float(lat), float(lon)) for lat, lon in polygon_latlon]
259    if len(pts) < 3:
260        return SurveyResult(False, message="A work area needs at least 3 points.")
261    if spacing_m is None or spacing_m < MIN_SPACING_M:
262        return SurveyResult(False, message=f"Spacing must be at least {MIN_SPACING_M:g} m.")
263
264    proj = Projection.for_point(pts[0][1], pts[0][0])
265    ring_m = [proj.point_to_metric(lon, lat) for lat, lon in pts]
266    poly_m = Polygon(ring_m)
267    if not poly_m.is_valid:
268        poly_m = poly_m.buffer(0)
269    if poly_m.is_empty or poly_m.area <= 0.0:
270        return SurveyResult(False, message="Degenerate work area.")
271
272    minx, miny, maxx, maxy = poly_m.bounds
273    # Pre-guard: refuse a too-small spacing before generating the grid.
274    if int((maxx - minx) / spacing_m + 1) * int((maxy - miny) / spacing_m + 1) > 50000:
275        return SurveyResult(False, message="Spacing too small for this area.")
276
277    rows: list[list[tuple[float, float]]] = []
278    y = miny + spacing_m / 2.0
279    ri = 0
280    while y <= maxy:
281        row = []
282        x = minx + spacing_m / 2.0
283        while x <= maxx:
284            if poly_m.contains(Point(x, y)):
285                row.append((x, y))
286            x += spacing_m
287        if ri % 2 == 1:
288            row.reverse()          # serpentine: alternate row direction
289        rows.append(row)
290        ri += 1
291        y += spacing_m
292
293    flat = [p for row in rows for p in row]
294    if not flat:
295        return SurveyResult(False, message="No spots fit inside the area at this spacing.")
296    if len(flat) > MAX_WORK_SPOTS:
297        return SurveyResult(
298            False,
299            message=f"{len(flat)} spots is too many; increase spacing (max {MAX_WORK_SPOTS}).",
300        )
301
302    waypoints = [
303        {"name": f"Spot {i + 1}", "lat": lat, "lon": lon}
304        for i, (x, y) in enumerate(flat)
305        for lon, lat in [proj.point_to_lonlat(x, y)]
306    ]
307    return SurveyResult(
308        True, waypoints=waypoints, spacing_m=spacing_m,
309        message=f"{len(waypoints)} spots on a {spacing_m:g} m grid.",
310    )
