vanchor.sim.battery

A simple simulated battery model (#60).

Trolling-motor autopilots run off a deep-cycle battery; how much charge is left (and how far the boat can still get on it) is a first-class safety concern. This model is intentionally lightweight -- it is not an electrochemical model -- but it is believable enough to drive the UI gauges and the Return-to-Launch auto-recommend logic (#61):

  • It draws a current that is the sum of a constant idle (hotel/standby) load and a propulsion load proportional to |thrust|. The propulsion draw is modelled at full thrust as load_a amps, scaled linearly with the thrust magnitude (a coarse but monotonic stand-in for the real cubic-ish curve).
  • It integrates the drawn charge out of the pack over time, lowering the state-of-charge (SOC).
  • It estimates time-to-empty and remaining range from a recent average draw and the boat's speed-over-ground, so the figures track how the boat is actually being used rather than an instantaneous spike.

On real hardware the SOC / voltage / current come from a battery monitor (a shunt + gauge) over the HAL instead of being integrated here; the telemetry shape and the range/time estimates stay identical so the UI and the RTL logic do not care which source is wired in. See BatteryConfig.

 29# SOC is held above this floor so a fully drained pack still reports a sane
 30# (non-negative) value rather than going negative.
 31_SOC_MIN = 0.0
 32_SOC_MAX = 100.0

 35@dataclass
 36class BatteryConfig:
 37    """Battery pack sizing + the load model.
 38
 39    ``reserve_pct`` is the usable-charge reserve: range / time-to-empty are
 40    reported down to this reserve (not to a flat-dead pack), matching how a
 41    skipper plans "I must turn back while I still have my reserve".
 42    """
 43
 44    capacity_ah: float = 100.0  # pack capacity (amp-hours)
 45    nominal_v: float = 12.0  # nominal terminal voltage
 46    reserve_pct: float = 15.0  # usable-charge reserve (%) kept in hand
 47    idle_a: float = 0.5  # constant hotel/standby draw (A)
 48    load_a: float = 40.0  # propulsion draw at full |thrust| (A)
 49    # Recent-draw smoothing time constant (s) for the range/time estimate, so a
 50    # momentary thrust spike doesn't make the range figure jump around.
 51    draw_tau_s: float = 20.0

 54class Battery:
 55    """Integrates state-of-charge from the applied thrust and reports estimates.
 56
 57    Pure and synchronous (no I/O, no clock) so it can be stepped deterministically
 58    from the simulator and unit-tested. ``step(dt, thrust, sog_mps)`` advances it.
 59    """
 60
 61    def __init__(self, config: BatteryConfig | None = None, soc_pct: float = 100.0) -> None:
 62        self.config = config or BatteryConfig()
 63        self.soc_pct = max(_SOC_MIN, min(_SOC_MAX, soc_pct))
 64        # Last instantaneous draw (A) and the smoothed recent average (A).
 65        self.current_a = self.config.idle_a
 66        self._avg_current_a = self.config.idle_a
 67        # Smoothed recent speed-over-ground (m/s) for the range estimate.
 68        self._avg_sog_mps = 0.0
 69
 70    def set_soc(self, soc_pct: float) -> None:
 71        """Set/reset the state-of-charge (e.g. swapping a fresh battery in)."""
 72        self.soc_pct = max(_SOC_MIN, min(_SOC_MAX, float(soc_pct)))
 73
 74    def _draw_for(self, thrust: float) -> float:
 75        cfg = self.config
 76        return cfg.idle_a + cfg.load_a * min(1.0, abs(thrust))
 77
 78    def step(self, dt: float, thrust: float, sog_mps: float) -> None:
 79        """Advance the SOC by one step under the given thrust + speed.
 80
 81        ``thrust`` is the normalized applied thrust (-1..1); ``sog_mps`` is the
 82        boat's speed over ground in m/s (used only for the range estimate).
 83        """
 84        if dt <= 0.0:
 85            return
 86        cfg = self.config
 87        self.current_a = self._draw_for(thrust)
 88
 89        # Integrate charge out of the pack: Ah drawn = A * (dt / 3600).
 90        ah_drawn = self.current_a * (dt / 3600.0)
 91        if cfg.capacity_ah > 0.0:
 92            self.soc_pct = max(_SOC_MIN, self.soc_pct - 100.0 * ah_drawn / cfg.capacity_ah)
 93
 94        # Smooth the recent draw + speed for stable range/time estimates.
 95        alpha = dt / (cfg.draw_tau_s + dt)
 96        self._avg_current_a += (self.current_a - self._avg_current_a) * alpha
 97        self._avg_sog_mps += (max(0.0, sog_mps) - self._avg_sog_mps) * alpha
 98
 99    # -- Derived telemetry ---------------------------------------------- #
100    @property
101    def voltage_v(self) -> float:
102        """A crude terminal voltage that sags as the pack drains.
103
104        Lead-acid resting voltage runs roughly from ~12.7 V full to ~11.8 V
105        empty; we linearly interpolate over that span around ``nominal_v`` so the
106        UI voltage gauge moves. (Real hardware reports a measured voltage.)
107        """
108        full = self.config.nominal_v + 0.7
109        empty = self.config.nominal_v - 0.2
110        return empty + (full - empty) * (self.soc_pct / 100.0)
111
112    @property
113    def draw_w(self) -> float:
114        """Instantaneous power draw in watts."""
115        return self.current_a * self.voltage_v
116
117    @property
118    def _usable_ah(self) -> float:
119        """Amp-hours left above the reserve floor."""
120        cfg = self.config
121        usable_pct = max(0.0, self.soc_pct - cfg.reserve_pct)
122        return cfg.capacity_ah * usable_pct / 100.0
123
124    @property
125    def time_to_empty_s(self) -> float:
126        """Estimated seconds until the usable reserve is hit, at the recent draw.
127
128        Returns ``inf`` when there is effectively no draw (nothing to estimate
129        against)."""
130        if self._avg_current_a <= 1e-6:
131            return float("inf")
132        return self._usable_ah / self._avg_current_a * 3600.0
133
134    @property
135    def range_m(self) -> float:
136        """Estimated metres of travel left at the recent average draw + speed.
137
138        Returns 0 when the boat isn't making way (no usable distance estimate)."""
139        tte = self.time_to_empty_s
140        if self._avg_sog_mps <= 1e-3 or tte == float("inf"):
141            return 0.0
142        return self._avg_sog_mps * tte
143
144    def to_dict(self) -> dict:
145        tte = self.time_to_empty_s
146        return {
147            "soc_pct": round(self.soc_pct, 1),
148            "voltage_v": round(self.voltage_v, 2),
149            "current_a": round(self.current_a, 2),
150            "draw_w": round(self.draw_w, 1),
151            "range_m": round(self.range_m, 1),
152            # JSON has no infinity; surface "unknown" (no meaningful draw) as null.
153            "time_to_empty_s": None if tte == float("inf") else round(tte, 1),
154        }
