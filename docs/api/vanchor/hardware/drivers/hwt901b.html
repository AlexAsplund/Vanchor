vanchor.hardware.drivers.hwt901b

Pluggable compass driver for the WitMotion HWT901B-TTL 9-axis AHRS.

Wraps the external hwt901b library and adapts it to vanchor's device seam: a ~vanchor.hardware.interfaces.Sensor that emits HDM NMEA onto the bus (nmea.in), exactly like SimCompass/SerialCompass -- so the navigator, controller and every mode are unchanged. Registers itself as the compass source "hwt901b" (see vanchor.hardware.registry), so it needs no edit to app.py's build seam.

Two device-specific features:

  • Auto-declination (+ mount offset). The magnetometer reads magnetic heading; true heading needs the local declination, and a real install has a fixed mount misalignment. Rather than make the skipper type a number, the combined offset is learned by comparing the compass heading to the GPS course-over-ground on straight-line runs (HeadingOffsetEstimator) -- no magnetic-model data to ship, and the mount error is corrected for free.
  • Device menu. HWT901BCompass.device_menu() advertises device-specific settings + actions (declination mode, magnetometer calibration, profiling) the UI renders generically -- the pattern any future smart device follows.

The hwt901b library is optional and imported lazily, so the core install and the simulator never need it; the driver itself is fully testable with a fake sensor (no serial port).

 80class HeadingOffsetEstimator:
 81    """Learn the fixed offset (declination + compass mount error) between the
 82    magnetic heading and true north, from GPS course-over-ground.
 83
 84    While the boat runs roughly straight above ``min_sog_mps``,
 85    ``course_over_ground - magnetic_heading`` is that offset (plus noise). We
 86    low-pass it with a long time constant so turns and GPS jitter average out and
 87    only sustained straight-line agreement moves the estimate."""
 88
 89    def __init__(
 90        self, *, min_sog_mps: float = 0.8, time_constant_s: float = 25.0,
 91        max_turn_dps: float = 8.0,
 92    ) -> None:
 93        self.min_sog_mps = min_sog_mps
 94        self.time_constant_s = time_constant_s
 95        self.max_turn_dps = max_turn_dps
 96        self.offset_deg: float = 0.0
 97        self.settled: bool = False
 98        self._prev_cog: float | None = None
 99
100    def update(self, magnetic_heading_deg: float, cog_deg: float | None,
101               sog_mps: float | None, dt: float) -> float:
102        if cog_deg is None or sog_mps is None or sog_mps < self.min_sog_mps or dt <= 0:
103            self._prev_cog = cog_deg
104            return self.offset_deg
105        if self._prev_cog is not None:
106            if abs(angle_difference(self._prev_cog, cog_deg)) / dt > self.max_turn_dps:
107                self._prev_cog = cog_deg  # mid-turn -> skip (COG != heading here)
108                return self.offset_deg
109        self._prev_cog = cog_deg
110        target = angle_difference(magnetic_heading_deg, cog_deg)  # signed cog - mag
111        a = min(1.0, dt / self.time_constant_s)
112        self.offset_deg = normalize_deg(
113            self.offset_deg + a * angle_difference(self.offset_deg, target)
114        )
115        self.settled = True
116        return self.offset_deg

119class HWT901BCompass(Sensor):
120    """HWT901B AHRS presented as a vanchor NMEA compass sensor.
121
122    ``sensor`` is anything with ``read_true_heading(declination_deg, timeout)``
123    and ``close()`` -- the real :class:`hwt901b.HWT901B` or a fake in tests.
124    ``motion_provider`` feeds GPS (cog, sog) for auto-declination."""
125
126    def __init__(
127        self, sensor: Any, bus: EventBus | None = None, *,
128        hz: float = 5.0,
129        motion_provider: MotionProvider | None = None,
130        declination_mode: str = "auto",           # "auto" | "manual" | "off"
131        manual_declination_deg: float = 0.0,
132        read_timeout: float = 0.5,
133    ) -> None:
134        self._sensor = sensor
135        self.bus = bus
136        self.hz = max(0.5, hz)
137        self.motion_provider = motion_provider
138        self.declination_mode = declination_mode
139        self.manual_declination_deg = manual_declination_deg
140        self.read_timeout = read_timeout
141        self.estimator = HeadingOffsetEstimator()
142        self.last_heading_deg: float | None = None
143        self._task: asyncio.Task | None = None
144
145    async def start(self) -> None:
146        self._task = asyncio.ensure_future(self._loop())
147
148    async def stop(self) -> None:
149        if self._task is not None:
150            self._task.cancel()
151            self._task = None
152        close = getattr(self._sensor, "close", None)
153        if close is not None:
154            await asyncio.get_event_loop().run_in_executor(None, close)
155
156    def _current_declination(self, magnetic_heading: float, dt: float) -> float:
157        if self.declination_mode == "manual":
158            return self.manual_declination_deg
159        if self.declination_mode == "off":
160            return 0.0
161        cog = sog = None
162        if self.motion_provider is not None:
163            mv = self.motion_provider()
164            if mv is not None:
165                cog, sog = mv
166        return self.estimator.update(magnetic_heading, cog, sog, dt)
167
168    async def sample_once(self, dt: float) -> str | None:
169        """Read one heading and return the HDM sentence (or None on timeout). The
170        blocking serial read runs in a thread so it never stalls the loop."""
171        loop = asyncio.get_event_loop()
172        try:
173            magnetic = await loop.run_in_executor(
174                None, lambda: self._sensor.read_true_heading(0.0, self.read_timeout)
175            )
176        except Exception as exc:  # noqa: BLE001 - timeout/parse; keep the loop alive
177            logger.debug("hwt901b read: %s", exc)
178            return None
179        heading = normalize_deg(magnetic + self._current_declination(magnetic, dt))
180        self.last_heading_deg = heading
181        return nmea.encode_hdm(heading)
182
183    async def _loop(self) -> None:
184        period = 1.0 / self.hz
185        while True:
186            try:
187                sentence = await self.sample_once(period)
188                if sentence and self.bus is not None:
189                    await self.bus.publish(events.NMEA_IN, sentence)
190            except asyncio.CancelledError:
191                raise
192            except Exception:  # pragma: no cover - defensive
193                logger.exception("hwt901b compass loop error")
194            await asyncio.sleep(period)
195
196    # -- device-specific settings menu (rendered generically by the UI) ---- #
197    def device_menu(self) -> dict:
198        return _menu_schema(self.declination_mode, self.manual_declination_deg, self.hz)
199
200    def apply_setting(self, key: str, value: Any) -> dict:
201        if key == "declination_mode" and value in ("auto", "manual", "off"):
202            self.declination_mode = value
203        elif key == "manual_declination_deg":
204            self.manual_declination_deg = float(value)
205        elif key == "hz":
206            self.hz = max(0.5, float(value))
207        else:
208            return {"ok": False, "message": f"unknown setting {key!r}"}
209        return {"ok": True}
210
211    def run_action(self, name: str, params: dict | None = None) -> dict:
212        if name == "profile":
213            hdg = self.last_heading_deg
214            return {"ok": True, "message": "HWT901B live status.", "status": {
215                "heading_deg": round(hdg, 1) if hdg is not None else None,
216                "declination_mode": self.declination_mode,
217                "offset_deg": round(self.estimator.offset_deg, 1),
218                "offset_settled": self.estimator.settled,
219            }}
220        if name == "calibrate_mag":
221            return {"ok": False, "message": "Interactive mag calibration is coming next."}
222        return {"ok": False, "message": f"unknown action {name!r}"}

225def open_hwt901b_compass(
226    port: str, baudrate: int, bus: EventBus | None, *,
227    hz: float = 5.0, motion_provider: MotionProvider | None = None,
228    declination_mode: str = "auto", manual_declination_deg: float = 0.0,
229) -> HWT901BCompass:
230    """Open the HWT901B on ``port`` and wrap it. Lazily imports the optional
231    ``hwt901b`` library (clear error if missing)."""
232    try:
233        from hwt901b import HWT901B
234    except ImportError as exc:  # pragma: no cover - env-dependent
235        raise RuntimeError(
236            "compass_source='hwt901b' needs the hwt901b library: "
237            "pip install 'vanchor[hwt901b]' (or `pip install -e ../python-hwt901b-ttl[serial]`)"
238        ) from exc
239    return HWT901BCompass(
240        HWT901B.open(port, baudrate=baudrate), bus, hz=hz,
241        motion_provider=motion_provider, declination_mode=declination_mode,
242        manual_declination_deg=manual_declination_deg,
243    )

246def _build(runtime: Any, cfg: Any) -> HWT901BCompass:
247    """Registry build hook: create the driver wired to the runtime's GPS motion
248    (for auto-declination) and bus, applying persisted device-menu settings."""
249    hw = cfg.hardware
250
251    def motion():
252        st = getattr(runtime, "state", None)
253        if st is None or st.fix is None:
254            return None
255        return (st.fix.cog_deg, st.sog_knots * 0.514444)  # knots -> m/s
256
257    saved = (getattr(hw, "device_settings", None) or {}).get("compass", {})
258    return open_hwt901b_compass(
259        hw.compass_port, hw.baudrate, runtime.bus,
260        hz=float(saved.get("hz", cfg.sensors.compass_hz)),
261        motion_provider=motion,
262        declination_mode=str(saved.get("declination_mode", "auto")),
263        manual_declination_deg=float(saved.get("manual_declination_deg", 0.0)),
264    )
265
266
267register_driver("compass", "hwt901b", _build,
268                label="WitMotion HWT901B AHRS", menu=default_menu())
