vanchor.hardware.serial_link

Serial transport abstraction for real-hardware drivers.

The drivers in vanchor.hardware.serial_devices talk to physical GPS, compass and motor controllers over a serial line. To keep them testable with no physical port (and to keep the import graph free of a hard pyserial dependency), all byte-level I/O goes through a small line-oriented transport abstraction:

SerialTransport -- the interface: open/close + read_line/write_line FakeSerialTransport -- in-memory transport for tests; push inbound lines, inspect outbound lines PySerialTransport -- real transport backed by serial_asyncio; the import is guarded so importing this module never requires the serial extra to be installed

Lines are newline-delimited UTF-8 strings; the transport strips the trailing newline on read and appends \r\n on write (standard for NMEA / Arduino serial protocols).

 30class SerialTransport(abc.ABC):
 31    """A line-oriented, asynchronous serial transport.
 32
 33    Implementations move whole text lines to and from some underlying byte
 34    stream. Drivers depend only on this interface, so the same driver code runs
 35    against a real port (:class:`PySerialTransport`) or an in-memory fake
 36    (:class:`FakeSerialTransport`).
 37    """
 38
 39    @abc.abstractmethod
 40    async def open(self) -> None:
 41        """Open/connect the underlying stream (idempotent)."""
 42
 43    @abc.abstractmethod
 44    async def close(self) -> None:
 45        """Close the underlying stream (idempotent)."""
 46
 47    @abc.abstractmethod
 48    async def read_line(self) -> str:
 49        """Read one line, with the line terminator stripped.
 50
 51        Blocks (asynchronously) until a line is available. May raise
 52        :class:`asyncio.CancelledError` when the awaiting task is cancelled, or
 53        ``EOFError`` when the stream is closed.
 54        """
 55
 56    @abc.abstractmethod
 57    async def write_line(self, line: str) -> None:
 58        """Write one line; the terminator (``\\r\\n``) is appended here."""

 61class FakeSerialTransport(SerialTransport):
 62    """In-memory transport for deterministic tests.
 63
 64    Tests push inbound lines with :meth:`feed` (which a reader picks up via
 65    :meth:`read_line`) and inspect everything a driver wrote via the
 66    :attr:`written` list.
 67    """
 68
 69    def __init__(self) -> None:
 70        self._inbound: asyncio.Queue[str | None] = asyncio.Queue()
 71        self.written: list[str] = []
 72        self.opened: bool = False
 73        self.closed: bool = False
 74
 75    # -- test helpers ----------------------------------------------------- #
 76    def feed(self, line: str) -> None:
 77        """Make ``line`` available to the next :meth:`read_line` call."""
 78        self._inbound.put_nowait(line.rstrip("\r\n"))
 79
 80    def feed_eof(self) -> None:
 81        """Signal end-of-stream; a pending/next :meth:`read_line` raises EOF."""
 82        self._inbound.put_nowait(None)
 83
 84    # -- SerialTransport -------------------------------------------------- #
 85    async def open(self) -> None:
 86        self.opened = True
 87        self.closed = False
 88
 89    async def close(self) -> None:
 90        self.closed = True
 91
 92    async def read_line(self) -> str:
 93        line = await self._inbound.get()
 94        if line is None:
 95            raise EOFError("fake serial transport closed")
 96        return line
 97
 98    async def write_line(self, line: str) -> None:
 99        self.written.append(line)

102class PySerialTransport(SerialTransport):
103    """Real serial transport backed by ``pyserial-asyncio``.
104
105    The ``serial_asyncio`` import is deferred to :meth:`open` so that merely
106    importing this module never requires the optional ``serial`` extra or any
107    physical hardware. Construct it with a device ``port`` (e.g.
108    ``"/dev/ttyUSB0"``) and a ``baudrate``.
109    """
110
111    def __init__(self, port: str, baudrate: int = 4800) -> None:
112        self.port = port
113        self.baudrate = baudrate
114        self._reader: asyncio.StreamReader | None = None
115        self._writer: asyncio.StreamWriter | None = None
116
117    async def open(self) -> None:
118        if self._reader is not None:
119            return
120        try:
121            import serial_asyncio  # type: ignore[import-untyped]
122        except ImportError as exc:  # pragma: no cover - needs missing extra
123            raise RuntimeError(
124                "PySerialTransport requires the 'serial' extra "
125                "(pip install pyserial-asyncio)"
126            ) from exc
127        self._reader, self._writer = await serial_asyncio.open_serial_connection(
128            url=self.port, baudrate=self.baudrate
129        )
130        logger.info("opened serial port %s @ %d baud", self.port, self.baudrate)
131
132    async def close(self) -> None:
133        writer, self._writer = self._writer, None
134        self._reader = None
135        if writer is not None:  # pragma: no cover - needs real port
136            writer.close()
137            try:
138                await writer.wait_closed()
139            except Exception:  # pragma: no cover - defensive
140                logger.debug("error while closing serial port %s", self.port)
141
142    async def read_line(self) -> str:  # pragma: no cover - needs real port
143        if self._reader is None:
144            raise RuntimeError("transport not open")
145        raw = await self._reader.readline()
146        if not raw:
147            raise EOFError(f"serial port {self.port} closed")
148        return raw.decode("ascii", errors="replace").rstrip("\r\n")
149
150    async def write_line(self, line: str) -> None:  # pragma: no cover - real port
151        if self._writer is None:
152            raise RuntimeError("transport not open")
153        self._writer.write((line + "\r\n").encode("ascii", errors="replace"))
154        await self._writer.drain()
