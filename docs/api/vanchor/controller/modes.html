vanchor.controller.modes

Control modes -- the high-level steering behaviours.

Each mode is a pure strategy: given the current NavigationState and a timestep, produce a Setpoint. A mode never touches hardware; it only expresses intent (either drive the motor directly, or hold a target heading). The controller's helm turns a heading intent into actual steering, so all the guided modes share one well-tuned heading loop.

This separation makes every behaviour independently unit-testable with no hardware and no event loop.

 86class DriftMode(ControlMode):
 87    """Controlled drift: hold a heading while a bidirectional SOG PID holds a
 88    *low* target speed-over-ground (``state.drift_target_knots``).
 89
 90    Wind/current carry the boat along the chosen bearing; the motor only trims
 91    speed -- adding thrust if the drift is too slow, or reversing to brake if it
 92    is too fast. Mirrors the Drift mode of high-end GPS trolling motors.
 93    """
 94
 95    name = ControlModeName.DRIFT
 96
 97    def __init__(self, config: DriftConfig | None = None) -> None:
 98        self.config = config or DriftConfig()
 99        self.pid = PID(
100            kp=self.config.kp, ki=self.config.ki, kd=0.0, output_min=-1.0, output_max=1.0
101        )
102
103    def activate(self, state: NavigationState) -> None:
104        self.pid.reset()
105
106    def update(self, state: NavigationState, dt: float) -> Setpoint:
107        self.pid.setpoint = state.drift_target_knots
108        thrust = self.pid.update(state.sog_knots, dt)
109        return GuidedSetpoint(target_heading=state.target_heading, thrust=thrust)

112@dataclass
113class AnchorConfig:
114    kp: float = 0.12  # thrust per metre of position error
115    kd: float = 0.6  # *braking* thrust per (m/s) of closing speed toward the anchor
116    max_thrust: float = 1.0
117    # Idle dead-band: once within this distance of the mark, idle the motor.
118    # With ~1.5 m GPS noise this stops the boat hunting/oscillating around the
119    # mark (it idles and drifts within the band instead -- calmer, easier on the
120    # motor) -- and because the motor isn't running, the boat doesn't yaw, so its
121    # heading is held passively (it does not spin to chase the mark). Drift is
122    # still rejected: once pushed past the band the PD drives it back. 0.8 m
123    # roughly halves the settling time and oscillation vs. no dead-band (see
124    # docs/analysis.md) while staying 100% within a tight radius.
125    idle_deadband_m: float = 0.8
126    # Aggressive "recover" (re-point and drive back to the mark) only kicks in
127    # past this distance, regardless of how small the anchor radius is. Without
128    # it, a tight radius (1-2 m) sits below the GPS-noise + boat-length floor, so
129    # the boat is *always* "outside" and stuck darting in recover mode -- the
130    # classic small-radius overcorrection. With it, a small radius holds just as
131    # calmly as a large one (gentle station-keeping), only entering recover when
132    # genuinely shoved off station.
133    recover_floor_m: float = 3.5
134    # Optional low-pass (EMA) on the perceived position. 1.0 = off (filtering
135    # adds lag a control loop fights, so it is off by default).
136    pos_filter_alpha: float = 1.0
137
138    # --- Predictive drift handling --------------------------------------- #
139    # The drift is *estimated* (from true ground velocity minus our propulsion)
140    # and exposed for the HUD/telemetry. Because the GPS now reports real
141    # course/speed-over-ground, the kd (closing-speed) term already *anticipates*
142    # drift -- it acts on velocity, so it counters the drift before the position
143    # error builds, holding ~100% within radius with little maneuvering. The
144    # explicit feed-forward below then double-counts and over-corrects, so it is
145    # OFF by default; it remains available (e.g. for very low control rates).
146    feedforward: bool = False
147    feedforward_gain: float = 0.7  # fraction of the estimated drift to counter
148    boat_max_speed_mps: float = 1.6  # to estimate our thrust's contribution to v
149    drift_alpha: float = 0.02  # EMA gain/tick for the drift estimate (~10 s @ 5 Hz)
150    drift_min_mps: float = 0.05  # below this, no significant drift to point into

153class AnchorHoldMode(ControlMode):
154    """Virtual anchor ("Spot-Lock"): hold position with reverse thrust + braking.
155
156    A PD controller on the (ground) distance to the mark: ``kp`` pulls toward the
157    anchor, ``kd`` brakes using the GPS closing speed so the boat doesn't
158    overshoot and orbit. It uses **reverse thrust** -- braking an overshoot, and
159    when the anchor ends up *behind* the boat it backs straight up toward it
160    rather than looping all the way around. Within ``idle_deadband_m`` it idles;
161    with no thrust the (thrust-to-steer) motor produces no yaw, so the heading is
162    held passively and the servo isn't worked.
163
164    Note: a single bow-mounted, thrust-to-steer motor is underactuated -- it
165    cannot actively hold an arbitrary heading while sitting still (steering needs
166    thrust, which moves it off station). So this holds *position* and lets the
167    heading settle, exactly like a real GPS trolling motor.
168    """
169
170    name = ControlModeName.ANCHOR_HOLD
171
172    def __init__(self, config: AnchorConfig | None = None) -> None:
173        self.config = config or AnchorConfig()
174        self._reverse = False  # hysteresis on the forward/reverse decision
175        self._recovering = False  # hysteresis on the recover/station decision
176        self._filt: GeoPoint | None = None  # EMA-filtered perceived position
177        self._drift_e = 0.0  # estimated drift velocity (world east, m/s)
178        self._drift_n = 0.0  # estimated drift velocity (world north, m/s)
179
180    def activate(self, state: NavigationState) -> None:
181        self._reverse = False
182        self._recovering = False
183        self._filt = None
184        self._drift_e = 0.0
185        self._drift_n = 0.0
186
187    def _update_drift(self, state: NavigationState) -> None:
188        """Estimate the environmental drift velocity (world frame) by subtracting
189        our own propulsion from the observed GPS velocity, low-passed."""
190        fix = state.fix
191        if fix is None:
192            return
193        v = knots_to_mps(fix.sog_knots)
194        cog = math.radians(fix.cog_deg)
195        v_e, v_n = v * math.sin(cog), v * math.cos(cog)
196        # Our thrust roughly produces thrust*max_speed along the boat heading.
197        h = math.radians(state.heading_deg)
198        thr = state.motor_command.thrust
199        int_e = thr * self.config.boat_max_speed_mps * math.sin(h)
200        int_n = thr * self.config.boat_max_speed_mps * math.cos(h)
201        a = self.config.drift_alpha
202        self._drift_e += a * ((v_e - int_e) - self._drift_e)
203        self._drift_n += a * ((v_n - int_n) - self._drift_n)
204        state.est_drift_mps = math.hypot(self._drift_e, self._drift_n)
205        state.est_drift_dir = math.degrees(math.atan2(self._drift_e, self._drift_n)) % 360.0
206
207    def _filtered_position(self, raw: GeoPoint) -> GeoPoint:
208        a = self.config.pos_filter_alpha
209        if self._filt is None or a >= 1.0:
210            self._filt = raw
211        else:
212            self._filt = GeoPoint(
213                self._filt.lat + a * (raw.lat - self._filt.lat),
214                self._filt.lon + a * (raw.lon - self._filt.lon),
215            )
216        return self._filt
217
218    def _closing_speed(self, state: NavigationState, toward_deg: float) -> float:
219        """GPS-derived speed component toward ``toward_deg`` (+ = approaching)."""
220        fix = state.fix
221        if fix is None:
222            return 0.0
223        v = knots_to_mps(fix.sog_knots)
224        return v * math.cos(math.radians(angle_difference(fix.cog_deg, toward_deg)))
225
226    def update(self, state: NavigationState, dt: float) -> Setpoint:
227        raw = state.position
228        if raw is None or state.anchor is None:
229            return ManualSetpoint(0.0, 0.0)
230
231        pos = self._filtered_position(raw)
232        distance = haversine_m(pos, state.anchor)
233        bearing = initial_bearing(pos, state.anchor)
234        state.distance_to_anchor_m = distance
235        state.bearing_to_dest = bearing
236        cfg = self.config
237        radius = state.anchor_radius_m
238
239        self._update_drift(state)
240
241        # Recover (actively re-point at the mark) only when pushed clearly out of
242        # the radius; resume station-keeping once back well inside. The
243        # hysteresis stops GPS noise from toggling the two. The effective trigger
244        # never drops below recover_floor_m, so a tight radius doesn't put the
245        # boat in permanent recovery (overcorrection) against GPS noise.
246        recover_at = max(radius, cfg.recover_floor_m)
247        if distance > recover_at:
248            self._recovering = True
249        elif distance < recover_at * 0.6:
250            self._recovering = False
251
252        if self._recovering:
253            # Drive toward the anchor, forward if it's ahead else reverse (back
254            # straight up) -- whichever needs less turning -- braked by closing
255            # speed so we don't overshoot and orbit.
256            closing = self._closing_speed(state, bearing)
257            demand = _clamp(
258                cfg.kp * distance - cfg.kd * closing, -cfg.max_thrust, cfg.max_thrust
259            )
260            angle = angle_difference(state.heading_deg, bearing)
261            if abs(angle) > 110.0:
262                self._reverse = True
263            elif abs(angle) < 70.0:
264                self._reverse = False
265            if not self._reverse:
266                return GuidedSetpoint(target_heading=bearing, thrust=demand)
267            return GuidedSetpoint(
268                target_heading=normalize_deg(bearing - 180.0), thrust=-demand
269            )
270
271        # --- Station-keeping inside the radius --------------------------- #
272        drift_mag = math.hypot(self._drift_e, self._drift_n)
273        if cfg.feedforward and drift_mag >= cfg.drift_min_mps:
274            # Predictive: point the bow *into* the drift and hold a steady
275            # counter-thrust (feed-forward), so the boat sits still against
276            # wind/current instead of drifting out and darting back. The bow
277            # axis is the direction opposing the drift.
278            held = math.degrees(math.atan2(-self._drift_e, -self._drift_n)) % 360.0
279            feed_forward = cfg.feedforward_gain * drift_mag / cfg.boat_max_speed_mps
280        else:
281            # No significant drift learned yet: hold the current heading.
282            held = state.heading_deg
283            feed_forward = 0.0
284
285        # Position trim along the bow axis, with a dead-zone so GPS noise near
286        # the mark doesn't cause fidgeting (the feed-forward does the holding).
287        fwd_err = distance * math.cos(math.radians(angle_difference(held, bearing)))
288        fwd_err = math.copysign(max(0.0, abs(fwd_err) - cfg.idle_deadband_m), fwd_err)
289        fwd_speed = self._closing_speed(state, held)
290        trim = cfg.kp * fwd_err - cfg.kd * fwd_speed
291        thrust = _clamp(feed_forward + trim, -cfg.max_thrust, cfg.max_thrust)
292
293        if abs(thrust) < 0.01:
294            # Nothing to do: idle (no thrust => no yaw => heading held).
295            return GuidedSetpoint(target_heading=state.heading_deg, thrust=0.0)
296        return GuidedSetpoint(target_heading=held, thrust=thrust)

299def maneuver_to_bearing(
300    heading_deg: float,
301    bearing_deg: float,
302    distance_m: float,
303    *,
304    turn_rate_dps: float,
305    fwd_speed_mps: float,
306    reverse_efficiency: float,
307    currently_reverse: bool = False,
308    hysteresis: float = 0.85,
309) -> tuple[float, float, bool]:
310    """Pick **forward** (bow toward ``bearing``) or **reverse** (stern toward
311    ``bearing``) to reach a point ``distance_m`` away on ``bearing_deg``, by
312    whichever has the lower estimated *time to arrive*: ``turn_time + travel_time``.
313
314    Reversing trades a smaller heading change for slower travel (a trolling-motor
315    prop is weaker astern), so it wins when the target is **behind AND near** —
316    sometimes that means "turn a little and reverse" rather than swinging the
317    whole boat around; for a *far* target it is quicker to turn around and run
318    forward. ``hysteresis`` (<1) makes switching require the alternative be
319    clearly better, to stop chatter near the crossover.
320
321    Returns ``(target_heading_deg, thrust_sign, reverse)`` where ``thrust_sign``
322    is ``+1`` forward / ``-1`` reverse. (The helm flips steering authority under
323    negative thrust, so a reverse setpoint steers correctly.)
324    """
325    turn = max(turn_rate_dps, 1.0)
326    vf = max(fwd_speed_mps, 0.05)
327    vr = max(fwd_speed_mps * reverse_efficiency, 0.05)
328    a_fwd = abs(angle_difference(heading_deg, bearing_deg))
329    a_rev = abs(angle_difference(heading_deg, normalize_deg(bearing_deg - 180.0)))
330    cost_fwd = a_fwd / turn + distance_m / vf
331    cost_rev = a_rev / turn + distance_m / vr
332    if currently_reverse:
333        reverse = cost_fwd >= cost_rev * hysteresis  # stay reverse unless fwd clearly better
334    else:
335        reverse = cost_rev < cost_fwd * hysteresis   # switch to reverse only if clearly better
336    if reverse:
337        return normalize_deg(bearing_deg - 180.0), -1.0, True
338    return normalize_deg(bearing_deg), 1.0, False

357class WaypointMode(ControlMode):
358    """Steer through ``state.waypoints`` in order, correcting for cross-track
359    error so the boat tracks each leg rather than just aiming at the mark."""
360
361    name = ControlModeName.WAYPOINT
362
363    def __init__(self, config: WaypointConfig | None = None) -> None:
364        self.config = config or WaypointConfig()
365        self._leg_start: GeoPoint | None = None
366        self._reverse = False  # hysteresis on the forward/reverse decision
367        self._step = 1         # traversal direction: +1 forward, -1 back (patrol)
368
369    def activate(self, state: NavigationState) -> None:
370        self._leg_start = state.position
371        self._reverse = False
372        self._step = 1
373        state.route_complete = False
374
375    def _wrap_or_bounce(self, state: NavigationState, pos: GeoPoint) -> bool:
376        """``active_waypoint`` has run off an END of the route. ``route_loop``
377        wraps to the start and keeps circling; ``route_patrol`` reverses direction
378        and runs back the other way; a plain route completes. Returns True when
379        the route is now COMPLETE."""
380        n = len(state.waypoints)
381        if state.route_loop:
382            state.active_waypoint = 0
383            self._step = 1
384            self._leg_start = pos
385            return False
386        if state.route_patrol and n >= 2:
387            self._step = -self._step
388            state.active_waypoint += 2 * self._step  # off-the-end -> adjacent in-range mark
389            self._leg_start = pos
390            return False
391        state.route_complete = True
392        return True
393
394    def update(self, state: NavigationState, dt: float) -> Setpoint:
395        pos = state.position
396        if pos is None or not state.waypoints:
397            return ManualSetpoint(0.0, 0.0)
398
399        # Off either END of the route (>= len going forward, or < 0 going back in
400        # a patrol)? loop wraps, patrol reverses, a plain route completes.
401        if not 0 <= state.active_waypoint < len(state.waypoints):
402            if self._wrap_or_bounce(state, pos):
403                return GuidedSetpoint(target_heading=state.heading_deg, thrust=0.0)
404
405        if self._leg_start is None:
406            self._leg_start = pos
407
408        target = state.waypoints[state.active_waypoint].point
409        distance = haversine_m(pos, target)
410        state.distance_to_waypoint_m = distance
411
412        if distance <= self.config.arrival_radius_m:
413            # Arrived: advance to the next leg (in the current direction).
414            state.active_waypoint += self._step
415            self._leg_start = target
416            if not 0 <= state.active_waypoint < len(state.waypoints):
417                if self._wrap_or_bounce(state, pos):
418                    return GuidedSetpoint(target_heading=state.heading_deg, thrust=0.0)
419            target = state.waypoints[state.active_waypoint].point
420            distance = haversine_m(pos, target)
421            state.distance_to_waypoint_m = distance
422
423        bearing = initial_bearing(pos, target)
424        xte = cross_track(self._leg_start, target, pos)
425        state.cross_track_m = xte.distance_m
426        state.bearing_to_dest = bearing
427
428        # Forward or reverse? For a mark that's behind and near, backing toward it
429        # beats swinging the whole boat around (see maneuver_to_bearing).
430        if self.config.allow_reverse:
431            _, _, self._reverse = maneuver_to_bearing(
432                state.heading_deg, bearing, distance,
433                turn_rate_dps=self.config.turn_rate_dps,
434                fwd_speed_mps=self.config.boat_speed_mps,
435                reverse_efficiency=self.config.reverse_efficiency,
436                currently_reverse=self._reverse,
437            )
438        else:
439            self._reverse = False
440
441        if self._reverse:
442            # Back straight toward the mark (stern-first pure pursuit) — a short
443            # manoeuvre, so we don't layer cross-track trim on top. The helm
444            # flips steering authority under negative thrust.
445            return GuidedSetpoint(
446                target_heading=normalize_deg(bearing - 180.0),
447                thrust=-self.config.throttle,
448            )
449
450        correction = max(
451            -self.config.max_xte_correction_deg,
452            min(self.config.max_xte_correction_deg, self.config.xte_gain * xte.distance_m),
453        )
454        # Positive xte => boat is right of track => steer left => reduce heading.
455        heading = normalize_deg(bearing - correction)
456        return GuidedSetpoint(target_heading=heading, thrust=self.config.throttle)

471class WorkAreaMode(ControlMode):
472    """Work an area spot by spot: travel to ``state.waypoints[active]``, HOLD
473    position there (active spot-lock) while the user works, then advance to the
474    next spot -- after ``dwell_s`` ("timed" advance) and/or when the user taps
475    "Go to next spot" (``state.work_next_requested``). ``route_loop`` cycles the
476    spots; ``route_patrol`` runs them there-and-back; otherwise the boat holds the
477    final spot once the route is done.
478
479    Travel reuses the waypoint leg logic (cross-track + forward/reverse helm); the
480    hold delegates to AnchorHoldMode (spot-lock). Dwell time is accumulated from
481    ``dt`` so the deterministic harness drives it without a wall clock.
482    """
483
484    name = ControlModeName.WORK_AREA
485
486    def __init__(
487        self,
488        config: WorkAreaConfig | None = None,
489        *,
490        waypoint_config: WaypointConfig | None = None,
491        anchor_config: AnchorConfig | None = None,
492    ) -> None:
493        self.config = config or WorkAreaConfig()
494        self._travel = waypoint_config or WaypointConfig()  # shared leg-nav params
495        self._anchor = AnchorHoldMode(anchor_config)
496        self._phase = "travel"          # "travel" | "hold"
497        self._leg_start: GeoPoint | None = None
498        self._reverse = False
499        self._step = 1                  # +1 forward / -1 back (patrol)
500        self._dwell_elapsed = 0.0
501
502    def activate(self, state: NavigationState) -> None:
503        self._phase = "travel"
504        self._leg_start = state.position
505        self._reverse = False
506        self._step = 1
507        self._dwell_elapsed = 0.0
508        state.route_complete = False
509        state.work_holding = False
510        state.work_next_requested = False
511        state.work_dwell_remaining_s = 0.0
512
513    def _wrap_or_bounce(self, state: NavigationState, pos: GeoPoint) -> bool:
514        """Ran off an END of the spot list: loop wraps to the start, patrol
515        reverses direction, otherwise the work route completes (return True)."""
516        n = len(state.waypoints)
517        if state.route_loop:
518            state.active_waypoint = 0
519            self._step = 1
520            self._leg_start = pos
521            return False
522        if state.route_patrol and n >= 2:
523            self._step = -self._step
524            state.active_waypoint += 2 * self._step
525            self._leg_start = pos
526            return False
527        state.route_complete = True
528        return True
529
530    def _begin_hold(self, state: NavigationState, spot: GeoPoint, dt: float) -> Setpoint:
531        self._phase = "hold"
532        self._dwell_elapsed = 0.0
533        state.anchor = spot
534        state.anchor_heading = state.heading_deg
535        self._anchor.activate(state)
536        state.work_holding = True
537        return self._anchor.update(state, dt)
538
539    def update(self, state: NavigationState, dt: float) -> Setpoint:
540        pos = state.position
541        if pos is None or not state.waypoints:
542            state.work_holding = self._phase == "hold"
543            return ManualSetpoint(0.0, 0.0)
544
545        if not 0 <= state.active_waypoint < len(state.waypoints):
546            if self._wrap_or_bounce(state, pos):
547                state.work_holding = False
548                state.work_dwell_remaining_s = 0.0
549                return GuidedSetpoint(target_heading=state.heading_deg, thrust=0.0)
550
551        if self._phase == "hold":
552            sp = self._anchor.update(state, dt)  # spot-lock; sets distance_to_anchor_m
553            self._dwell_elapsed += dt
554            timed = self.config.advance == "timed"
555            want_advance = state.work_next_requested or (
556                timed and self._dwell_elapsed >= self.config.dwell_s
557            )
558            state.work_next_requested = False  # consume the button press
559            if want_advance and not state.route_complete:
560                self._leg_start = state.waypoints[state.active_waypoint].point
561                self._reverse = False
562                state.active_waypoint += self._step
563                if not 0 <= state.active_waypoint < len(state.waypoints):
564                    self._wrap_or_bounce(state, pos)  # loop/patrol, or sets route_complete
565                if state.route_complete:
566                    state.active_waypoint = max(
567                        0, min(state.active_waypoint, len(state.waypoints) - 1)
568                    )
569                else:
570                    self._phase = "travel"
571                    self._dwell_elapsed = 0.0
572                    state.work_holding = False
573                    state.work_dwell_remaining_s = 0.0
574                    # fall through to the travel leg below
575            if self._phase == "hold":
576                state.work_holding = True
577                state.work_dwell_remaining_s = (
578                    max(0.0, self.config.dwell_s - self._dwell_elapsed)
579                    if (timed and not state.route_complete) else 0.0
580                )
581                # On station, orient to the spot's desired heading if one is set.
582                # Best-effort: a single bow thruster can't perfectly hold heading
583                # AND position, so position recovery (the anchor) wins when the boat
584                # drifts out of the hold radius.
585                spot_hdg = state.waypoints[state.active_waypoint].heading
586                if spot_hdg is not None and state.distance_to_anchor_m <= max(state.anchor_radius_m, 1.0):
587                    return GuidedSetpoint(
588                        target_heading=spot_hdg % 360.0, thrust=self.config.orient_thrust
589                    )
590                return sp
591
592        # Travel toward the active spot.
593        if self._leg_start is None:
594            self._leg_start = pos
595        target = state.waypoints[state.active_waypoint].point
596        distance = haversine_m(pos, target)
597        state.distance_to_waypoint_m = distance
598        if distance <= self.config.arrival_radius_m:
599            return self._begin_hold(state, target, dt)
600
601        bearing = initial_bearing(pos, target)
602        xte = cross_track(self._leg_start, target, pos)
603        state.cross_track_m = xte.distance_m
604        state.bearing_to_dest = bearing
605        if self._travel.allow_reverse:
606            _, _, self._reverse = maneuver_to_bearing(
607                state.heading_deg, bearing, distance,
608                turn_rate_dps=self._travel.turn_rate_dps,
609                fwd_speed_mps=self._travel.boat_speed_mps,
610                reverse_efficiency=self._travel.reverse_efficiency,
611                currently_reverse=self._reverse,
612            )
613        else:
614            self._reverse = False
615        if self._reverse:
616            return GuidedSetpoint(
617                target_heading=normalize_deg(bearing - 180.0),
618                thrust=-self.config.throttle,
619            )
620        correction = max(
621            -self._travel.max_xte_correction_deg,
622            min(self._travel.max_xte_correction_deg, self._travel.xte_gain * xte.distance_m),
623        )
624        heading = normalize_deg(bearing - correction)
625        return GuidedSetpoint(target_heading=heading, thrust=self.config.throttle)

676class ContourFollowMode(ControlMode):
677    """Follow a depth contour (isobath).
678
679    Drives forward at the set speed while steering to keep ``state.depth_m`` at
680    ``target_depth_m``. It estimates the bottom's slope *along the track* from
681    the depth TREND -- how much the depth changed over the last few metres of
682    travel -- and curves toward deeper or shallower water to null the depth
683    error, so the boat tracks the chosen isobath rather than just driving
684    straight. ``side`` ("deep"/"shallow") picks which way it turns to correct,
685    matching the bank the operator wants to favour. If no sounding is available
686    it simply holds heading.
687    """
688
689    name = ControlModeName.CONTOUR_FOLLOW
690
691    def __init__(self, config: ContourConfig | None = None) -> None:
692        self.config = config or ContourConfig()
693        self._ref_pos: GeoPoint | None = None
694        self._ref_depth: float = 0.0
695        self._base_heading: float = 0.0  # along-contour heading we weave around
696        self.error_m = 0.0  # exposed for telemetry
697
698    def activate(self, state: NavigationState) -> None:
699        self._ref_pos = state.position
700        self._ref_depth = state.depth_m
701        self._base_heading = state.heading_deg
702        self.error_m = 0.0
703
704    def update(self, state: NavigationState, dt: float) -> Setpoint:
705        depth = state.depth_m
706        heading = state.heading_deg
707        thrust = self.config.throttle
708
709        # No usable sounding (<=0 = unknown/no return): hold heading, keep moving.
710        if depth <= 0.0:
711            return GuidedSetpoint(target_heading=heading, thrust=thrust)
712
713        target = state.contour_target_depth_m
714        # Positive error => we are too DEEP (need to head toward shallower water).
715        self.error_m = depth - target
716
717        pos = state.position
718        if pos is not None:
719            if self._ref_pos is None:
720                self._ref_pos = pos
721                self._ref_depth = depth
722            elif haversine_m(self._ref_pos, pos) >= self.config.trend_distance_m:
723                # Use the depth TREND along the track to keep the along-contour
724                # base heading aligned with the isobath as the bottom curves. If
725                # the depth changed while running near the target depth, the
726                # contour is bending, so rotate the base heading to chase it
727                # (toward deeper if we've drifted shallow, toward shallower if
728                # deep) -- a slow correction that follows a curving isobath.
729                ddepth = depth - self._ref_depth
730                if abs(self.error_m) < 1.0 and abs(ddepth) > 0.05:
731                    deep_side = 1.0 if state.contour_side == "deep" else -1.0
732                    nudge = deep_side * _clamp(-ddepth * 5.0, -15.0, 15.0)
733                    self._base_heading = normalize_deg(self._base_heading + nudge)
734                self._ref_pos = pos
735                self._ref_depth = depth
736
737        # ``side`` says which side of the boat the operator wants the DEEP water
738        # on. To null the depth error we weave off the along-contour BASE heading
739        # (captured when engaged), not the live heading -- otherwise a constant
740        # turn offset would just spin the boat in a circle:
741        #   too deep  (error>0) -> aim toward the SHALLOW side
742        #   too shallow(error<0)-> aim toward the DEEP side
743        # With "deep" on starboard, the deep side is +90deg; aiming toward deep
744        # is a positive (starboard) offset from base. "shallow" mirrors it. The
745        # magnitude is proportional to the error and capped so the approach to
746        # the isobath stays gentle.
747        deep_side_sign = 1.0 if state.contour_side == "deep" else -1.0
748        want_deeper = self.error_m < 0.0
749        turn = deep_side_sign if want_deeper else -deep_side_sign
750
751        magnitude = min(
752            self.config.max_offset_deg,
753            self.config.heading_gain_deg_per_m * abs(self.error_m),
754        )
755        offset = turn * magnitude
756        return GuidedSetpoint(
757            target_heading=normalize_deg(self._base_heading + offset), thrust=thrust
758        )

766    radial_gain_deg_per_m: float = 3.0
767    max_radial_correction_deg: float = 60.0

770class OrbitMode(ControlMode):
771    """Orbit a centre point at a fixed radius (circle / racetrack hold).
772
773    Each tick it computes the bearing from the centre to the boat, advances that
774    bearing a little in the travel direction (cw/ccw) to get a point slightly
775    *ahead* on the ring, aims there, and biases the heading by a radial-error
776    correction so the boat both converges to the ring and holds it. Drives
777    forward at the set speed. ``range_m`` (distance to centre) is exposed for
778    telemetry.
779    """
780
781    name = ControlModeName.ORBIT
782
783    def __init__(self, config: OrbitConfig | None = None) -> None:
784        self.config = config or OrbitConfig()
785        self.range_m = 0.0  # exposed for telemetry
786
787    def update(self, state: NavigationState, dt: float) -> Setpoint:
788        pos = state.position
789        center = state.orbit_center
790        if pos is None or center is None:
791            return GuidedSetpoint(target_heading=state.heading_deg, thrust=0.0)
792
793        radius = max(1.0, state.orbit_radius_m)
794        ccw = state.orbit_direction == "ccw"
795        sign = -1.0 if ccw else 1.0  # cw advances bearing-from-centre +, ccw -
796
797        range_m = haversine_m(center, pos)
798        self.range_m = range_m
799        state.distance_to_anchor_m = range_m  # reuse the HUD range field
800
801        bearing_out = initial_bearing(center, pos)  # centre -> boat
802
803        # Pure tangent: travel direction along the ring at the boat's bearing
804        # (perpendicular to the radial, on the chosen turn side).
805        tangent = normalize_deg(bearing_out + sign * 90.0)
806
807        # Radial correction so the boat converges to the ring and holds it:
808        # outside (radial_err>0) -> steer inward; inside -> steer outward. The
809        # correction rotates the tangent heading toward/away from the centre.
810        radial_err = range_m - radius  # + = outside
811        correction = _clamp(
812            self.config.radial_gain_deg_per_m * radial_err,
813            -self.config.max_radial_correction_deg,
814            self.config.max_radial_correction_deg,
815        )
816        # When outside (radial_err>0) steer toward the centre; when inside, steer
817        # outward. Rotating the tangent by ``+sign*correction`` turns it toward
818        # the inward radial for both cw and ccw (verified for each sign), so the
819        # boat spirals onto the ring and then holds the tangent (correction->0).
820        heading = normalize_deg(tangent + sign * correction)
821        return GuidedSetpoint(target_heading=heading, thrust=self.config.throttle)

119    # With ~1.5 m GPS noise this stops the boat hunting/oscillating around the

130    # the boat is *always* "outside" and stuck darting in recover mode -- the

301    heading_deg: float,
302    bearing_deg: float,
303    distance_m: float,
304    *,
305    turn_rate_dps: float,
306    fwd_speed_mps: float,

311    """Pick **forward** (bow toward ``bearing``) or **reverse** (stern toward
312    ``bearing``) to reach a point ``distance_m`` away on ``bearing_deg``, by
313    whichever has the lower estimated *time to arrive*: ``turn_time + travel_time``.
314
315    Reversing trades a smaller heading change for slower travel (a trolling-motor
316    prop is weaker astern), so it wins when the target is **behind AND near** —
317    sometimes that means "turn a little and reverse" rather than swinging the
318    whole boat around; for a *far* target it is quicker to turn around and run
319    forward. ``hysteresis`` (<1) makes switching require the alternative be
320    clearly better, to stop chatter near the crossover.
321

323    is ``+1`` forward / ``-1`` reverse. (The helm flips steering authority under
324    negative thrust, so a reverse setpoint steers correctly.)
325    """
326    turn = max(turn_rate_dps, 1.0)
327    vf = max(fwd_speed_mps, 0.05)
328    vr = max(fwd_speed_mps * reverse_efficiency, 0.05)
329    a_fwd = abs(angle_difference(heading_deg, bearing_deg))
330    a_rev = abs(angle_difference(heading_deg, normalize_deg(bearing_deg - 180.0)))
331    cost_fwd = a_fwd / turn + distance_m / vf
332    cost_rev = a_rev / turn + distance_m / vr
333    if currently_reverse:

335    else:
336        reverse = cost_rev < cost_fwd * hysteresis   # switch to reverse only if clearly better

719        if pos is not None:
720            if self._ref_pos is None:
721                self._ref_pos = pos
722                self._ref_depth = depth
723            elif haversine_m(self._ref_pos, pos) >= self.config.trend_distance_m:
724                # Use the depth TREND along the track to keep the along-contour
725                # base heading aligned with the isobath as the bottom curves. If
726                # the depth changed while running near the target depth, the
727                # contour is bending, so rotate the base heading to chase it
728                # (toward deeper if we've drifted shallow, toward shallower if
729                # deep) -- a slow correction that follows a curving isobath.
730                ddepth = depth - self._ref_depth
731                if abs(self.error_m) < 1.0 and abs(ddepth) > 0.05:
732                    deep_side = 1.0 if state.contour_side == "deep" else -1.0
733                    nudge = deep_side * _clamp(-ddepth * 5.0, -15.0, 15.0)
734                    self._base_heading = normalize_deg(self._base_heading + nudge)
735                self._ref_pos = pos
736                self._ref_depth = depth
737
738        # ``side`` says which side of the boat the operator wants the DEEP water
739        # on. To null the depth error we weave off the along-contour BASE heading
740        # (captured when engaged), not the live heading -- otherwise a constant
741        # turn offset would just spin the boat in a circle:
742        #   too deep  (error>0) -> aim toward the SHALLOW side
743        #   too shallow(error<0)-> aim toward the DEEP side
744        # With "deep" on starboard, the deep side is +90deg; aiming toward deep
745        # is a positive (starboard) offset from base. "shallow" mirrors it. The
746        # magnitude is proportional to the error and capped so the approach to
747        # the isobath stays gentle.
748        deep_side_sign = 1.0 if state.contour_side == "deep" else -1.0
749        want_deeper = self.error_m < 0.0
750        turn = deep_side_sign if want_deeper else -deep_side_sign
751
752        magnitude = min(
753            self.config.max_offset_deg,
754            self.config.heading_gain_deg_per_m * abs(self.error_m),
755        )
756        offset = turn * magnitude
757        return GuidedSetpoint(
758            target_heading=normalize_deg(self._base_heading + offset), thrust=thrust
759        )

772    """Orbit a centre point at a fixed radius (circle / racetrack hold).
773
774    Each tick it computes the bearing from the centre to the boat, advances that
775    bearing a little in the travel direction (cw/ccw) to get a point slightly
776    *ahead* on the ring, aims there, and biases the heading by a radial-error
777    correction so the boat both converges to the ring and holds it. Drives
778    forward at the set speed. ``range_m`` (distance to centre) is exposed for
779    telemetry.
780    """
781
782    name = ControlModeName.ORBIT
783
784    def __init__(self, config: OrbitConfig | None = None) -> None:
785        self.config = config or OrbitConfig()
786        self.range_m = 0.0  # exposed for telemetry
787
788    def update(self, state: NavigationState, dt: float) -> Setpoint:
789        pos = state.position
790        center = state.orbit_center
791        if pos is None or center is None:
792            return GuidedSetpoint(target_heading=state.heading_deg, thrust=0.0)
793
794        radius = max(1.0, state.orbit_radius_m)
795        ccw = state.orbit_direction == "ccw"
796        sign = -1.0 if ccw else 1.0  # cw advances bearing-from-centre +, ccw -
797
798        range_m = haversine_m(center, pos)
799        self.range_m = range_m
800        state.distance_to_anchor_m = range_m  # reuse the HUD range field
801
802        bearing_out = initial_bearing(center, pos)  # centre -> boat
803
804        # Pure tangent: travel direction along the ring at the boat's bearing
805        # (perpendicular to the radial, on the chosen turn side).
806        tangent = normalize_deg(bearing_out + sign * 90.0)
807
808        # Radial correction so the boat converges to the ring and holds it:
809        # outside (radial_err>0) -> steer inward; inside -> steer outward. The
810        # correction rotates the tangent heading toward/away from the centre.
811        radial_err = range_m - radius  # + = outside
812        correction = _clamp(
813            self.config.radial_gain_deg_per_m * radial_err,
814            -self.config.max_radial_correction_deg,
815            self.config.max_radial_correction_deg,
816        )
817        # When outside (radial_err>0) steer toward the centre; when inside, steer
818        # outward. Rotating the tangent by ``+sign*correction`` turns it toward
819        # the inward radial for both cw and ccw (verified for each sign), so the
820        # boat spirals onto the ring and then holds the tangent (correction->0).
821        heading = normalize_deg(tangent + sign * correction)
822        return GuidedSetpoint(target_heading=heading, thrust=self.config.throttle)
