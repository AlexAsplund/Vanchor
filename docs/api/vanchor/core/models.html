vanchor.core.models

Typed data models shared across the whole system.

These are deliberately small, immutable-ish dataclasses. They are the common vocabulary spoken by sensors, the navigator, control modes, the helm and the motor controller, so that real and simulated devices are interchangeable.

123@dataclass
124class Environment:
125    """Wind and current acting on the boat. Directions are *toward* which the
126    flow pushes, in degrees. Speeds are in m/s."""
127
128    current_speed: float = 0.0
129    current_dir: float = 0.0
130    wind_speed: float = 0.0
131    wind_dir: float = 0.0
132    # Fraction of wind speed that translates into hull drift (leeway).
133    wind_leeway: float = 0.03
134    # Gustiness: std (m/s) and correlation time (s) of the time-varying gust the
135    # simulator layers on top of the base wind. 0 amplitude = steady wind.
136    gust_amplitude_mps: float = 0.0
137    gust_tau_s: float = 5.0
138    # Slow weather wander (much slower than gusts), in [0, 1]. 0 = steady.
139    # The simulator evolves wind speed/direction (and current) by this much and
140    # writes the evolving values back into wind_speed/wind_dir/current_speed.
141    wind_variability: float = 0.0
142    current_variability: float = 0.0
143
144    def drift_vector(self) -> tuple[float, float]:
145        """Net environmental drift as an (east, north) velocity in m/s."""
146        ce = self.current_speed * math.sin(math.radians(self.current_dir))
147        cn = self.current_speed * math.cos(math.radians(self.current_dir))
148        we = self.wind_speed * self.wind_leeway * math.sin(math.radians(self.wind_dir))
149        wn = self.wind_speed * self.wind_leeway * math.cos(math.radians(self.wind_dir))
150        return (ce + we, cn + wn)

153@dataclass
154class BoatState:
155    """Ground-truth physical state of the (simulated) boat."""
156
157    point: GeoPoint = field(default_factory=lambda: GeoPoint(0.0, 0.0))
158    heading_deg: float = 0.0  # the way the bow points
159    speed_mps: float = 0.0  # forward speed through the water
160    timestamp: float = 0.0
161    # Velocity over ground (world frame, m/s) -- hull motion plus environmental
162    # drift. This is what a real GPS reports as course/speed over ground.
163    ground_ve: float = 0.0  # east
164    ground_vn: float = 0.0  # north
165

167def _clamp(value: float, low: float, high: float) -> float:
168    return max(low, min(high, value))
