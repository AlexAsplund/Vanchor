vanchor.core.state

The single shared navigation state.

This replaces the old project's stringly-typed nested-dict "DataNode". It is a plain typed object that the navigator writes to and that control modes read from. to_dict produces the telemetry payload streamed to the UI.

 22@dataclass
 23class NavigationState:
 24    """Everything the controller knows about the world *as reported by sensors*.
 25
 26    Crucially this is the boat's *perceived* state (from GPS/compass), not the
 27    simulator's ground truth -- the controller only ever steers on what the
 28    sensors tell it, exactly as it would with real hardware.
 29    """
 30
 31    # Latest perceived position / motion.
 32    fix: GpsFix | None = None
 33    fix_seq: int = 0  # bumped by the navigator on every fresh fix (freshness)
 34    heading_deg: float = 0.0  # latest compass heading
 35    sog_knots: float = 0.0  # speed over ground from GPS
 36    depth_m: float = 0.0  # water depth under the boat (from a depth sounder)
 37
 38    # Sensor-anomaly protection: how many implausible readings were rejected.
 39    heading_rejected: int = 0
 40    position_rejected: int = 0
 41
 42    # Active behaviour and its parameters.
 43    mode: ControlModeName = ControlModeName.MANUAL
 44    anchor: GeoPoint | None = None
 45    anchor_radius_m: float = 5.0
 46    anchor_heading: float = 0.0  # heading to hold while station-keeping
 47    target_heading: float = 0.0
 48    drift_target_knots: float = 0.5  # target speed-over-ground for Drift mode
 49
 50    # --- Contour-follow (#57): hold a depth contour (isobath). ----------- #
 51    contour_target_depth_m: float = 0.0
 52    contour_side: str = "deep"  # "deep" | "shallow" -- which way to turn
 53
 54    # --- Circle / orbit (#58): orbit a centre at a fixed radius. ---------- #
 55    orbit_center: GeoPoint | None = None
 56    orbit_radius_m: float = 20.0
 57    orbit_direction: str = "cw"  # "cw" | "ccw"
 58
 59    # --- Trolling pattern (#59): sinusoidal S-curve weave. --------------- #
 60    trolling_base_heading: float = 0.0
 61    trolling_amplitude_deg: float = 20.0
 62    trolling_period_s: float = 20.0
 63
 64    waypoints: list[Waypoint] = field(default_factory=list)
 65    active_waypoint: int = 0
 66    route_on_arrival: str = "none"  # "anchor" | "stop" | "none" when route done
 67    route_complete: bool = False
 68    # When True, reaching the last waypoint wraps back to the first (closed loop,
 69    # e.g. the "around island" route) instead of completing -- the boat circles
 70    # continuously. Set via the goto/load_route "loop" flag.
 71    route_loop: bool = False
 72    # When True, reaching either END of the route reverses direction and runs it
 73    # back the other way -- a continuous there-and-back "patrol" (distinct from
 74    # route_loop, which closes the ring). Set via the goto/load_route "patrol" flag.
 75    route_patrol: bool = False
 76
 77    # --- Work Area mode: visit spots, hold at each, then advance. -------- #
 78    # The spots are state.waypoints; active_waypoint is the current spot. While
 79    # holding, the UI shows a big "Go to next spot" button (driven by work_holding).
 80    work_holding: bool = False             # currently spot-locked at a spot
 81    work_dwell_remaining_s: float = 0.0    # countdown to auto-advance (timed mode)
 82    work_next_requested: bool = False      # transient: the "next spot" button press
 83
 84    # --- Return-to-Launch (#61): the recorded home/launch point. --------- #
 85    launch: GeoPoint | None = None  # first good fix, or set via set_launch
 86    rtl_recommended: bool = False  # battery can *just* make it home -> UI prompt
 87
 88    # --- Man-overboard (#63): the MOB mark to return to. ----------------- #
 89    mob: GeoPoint | None = None
 90    mob_active: bool = False
 91
 92    # Latest actuator command produced by the controller.
 93    motor_command: MotorCommand = field(default_factory=MotorCommand)
 94    # Physical steer range of the trolling motor (deg each side of the bow). The
 95    # real motor rotates a bit past 360° total (cable-limited); this maps the
 96    # normalized steering command to a real azimuth for display/telemetry.
 97    max_steer_angle_deg: float = 35.0
 98
 99    # Diagnostics surfaced to the UI so a human can see *why* the controller
100    # is doing what it is doing.
101    distance_to_anchor_m: float = 0.0
102    distance_to_waypoint_m: float = 0.0
103    cross_track_m: float = 0.0
104    bearing_to_dest: float = 0.0
105    last_apb: str | None = None
106
107    # Anchor controller's learned environmental drift (for display / feed-forward).
108    est_drift_mps: float = 0.0
109    est_drift_dir: float = 0.0  # degrees the drift pushes toward
110
111    # Latest parsed APB (decomposed to keep core/ free of nav imports) used by
112    # the external-autopilot FollowAPB mode.
113    has_apb: bool = False
114    apb_cross_track_m: float = 0.0
115    apb_steer_to: str = "L"
116    apb_bearing_to_dest: float = 0.0
117
118    @property
119    def position(self) -> GeoPoint | None:
120        return self.fix.point if self.fix else None
121
122    def to_dict(self) -> dict:
123        pos = self.position
124        return {
125            "mode": self.mode.value,
126            "position": {"lat": pos.lat, "lon": pos.lon} if pos else None,
127            "heading_deg": round(self.heading_deg, 2),
128            "sog_knots": round(self.sog_knots, 2),
129            "depth_m": round(self.depth_m, 1),
130            "sensors": {
131                "heading_rejected": self.heading_rejected,
132                "position_rejected": self.position_rejected,
133            },
134            "anchor": (
135                {"lat": self.anchor.lat, "lon": self.anchor.lon}
136                if self.anchor
137                else None
138            ),
139            "anchor_radius_m": self.anchor_radius_m,
140            "anchor_heading": round(self.anchor_heading, 2),
141            "target_heading": round(self.target_heading, 2),
142            "drift_target_knots": round(self.drift_target_knots, 2),
143            "waypoints": [
144                {"name": w.name, "lat": w.point.lat, "lon": w.point.lon,
145                 "heading": w.heading}
146                for w in self.waypoints
147            ],
148            "active_waypoint": self.active_waypoint,
149            "route_loop": self.route_loop,
150            "route_patrol": self.route_patrol,
151            "work_holding": self.work_holding,
152            "work_dwell_remaining_s": round(self.work_dwell_remaining_s, 1),
153            "work_spot_count": len(self.waypoints),
154            "motor": {
155                **asdict(self.motor_command),
156                # Trolling-motor azimuth relative to the bow (deg), for the UI
157                # motor-direction indicator. Reverse thrust points it astern.
158                "steer_angle_deg": round(
159                    self.motor_command.steering * self.max_steer_angle_deg
160                    + (180.0 if self.motor_command.thrust < 0 else 0.0),
161                    1,
162                ),
163            },
164            "distance_to_anchor_m": round(self.distance_to_anchor_m, 2),
165            "distance_to_waypoint_m": round(self.distance_to_waypoint_m, 2),
166            "cross_track_m": round(self.cross_track_m, 2),
167            "bearing_to_dest": round(self.bearing_to_dest, 2),
168            "est_drift_mps": round(self.est_drift_mps, 3),
169            "est_drift_dir": round(self.est_drift_dir, 1),
170            "last_apb": self.last_apb,
171            "launch": {
172                "lat": self.launch.lat if self.launch else None,
173                "lon": self.launch.lon if self.launch else None,
174                "set": self.launch is not None,
175            },
176            "rtl_recommended": self.rtl_recommended,
177            "mob": {
178                "active": self.mob_active,
179                "lat": self.mob.lat if self.mob else None,
180                "lon": self.mob.lon if self.mob else None,
181            },
182        }
