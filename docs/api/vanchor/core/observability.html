vanchor.core.observability

Logging setup, telemetry recording, and an event-bus wiretap.

Three small, independent observability helpers:

  • setup_logging() configures the root logger consistently for the app and tests.
  • TelemetryRecorder keeps an in-memory ring of recent telemetry snapshots and (optionally) appends each one as a JSON line to a file, so a run can be replayed or inspected after the fact.
  • wiretap() attaches a wildcard subscriber to the EventBus that logs every (topic, payload) at DEBUG -- a cheap, central trace of the whole system's message flow.
  • DecisionLog is an optional ring buffer answering "why did the controller do that" by recording human-readable reasons with fields.

Everything here is deliberately defensive: recording must never crash the control loop, so serialization failures and file errors are swallowed and logged rather than propagated.

 38class RingLogHandler(logging.Handler):
 39    """Keep the most recent log records in memory for the "View logs" UI.
 40
 41    A bounded ring of decoded records (time/level/logger/message). Cheap and
 42    crash-safe; :func:`log_ring` returns the process-wide singleton so the buffer
 43    survives repeated :func:`setup_logging` calls."""
 44
 45    def __init__(self, capacity: int = 800) -> None:
 46        super().__init__()
 47        self.records: Deque[dict] = deque(maxlen=capacity)
 48
 49    def emit(self, record: logging.LogRecord) -> None:
 50        try:
 51            self.records.append({
 52                "t": record.created,
 53                "level": record.levelname,
 54                "levelno": record.levelno,
 55                "name": record.name,
 56                "msg": record.getMessage(),
 57            })
 58        except Exception:  # pragma: no cover - logging must never raise
 59            pass
 60
 61    def dump(self, min_levelno: int = 0, limit: int = 500,
 62             contains: str | None = None) -> list[dict]:
 63        """Newest-last records at/above ``min_levelno``, optionally text-filtered."""
 64        rows = [r for r in self.records if r["levelno"] >= min_levelno]
 65        if contains:
 66            needle = contains.lower()
 67            rows = [r for r in rows if needle in r["msg"].lower() or needle in r["name"].lower()]
 68        return rows[-limit:]

 83def setup_logging(level: str = "INFO", fmt: str | None = None) -> None:
 84    """Configure the root logger with a single stream handler.
 85
 86    Safe to call more than once: existing handlers are cleared first so we do
 87    not accumulate duplicate handlers (and duplicate log lines) across calls.
 88
 89    ``level`` is a standard level name (``"DEBUG"``, ``"INFO"`` ...). Unknown
 90    names fall back to ``INFO``.
 91    """
 92    numeric = logging.getLevelName(level.upper())
 93    if not isinstance(numeric, int):
 94        numeric = logging.INFO
 95
 96    root = logging.getLogger()
 97    # Drop any handlers a previous setup (or basicConfig) installed.
 98    for handler in list(root.handlers):
 99        root.removeHandler(handler)
100
101    handler = logging.StreamHandler()
102    handler.setFormatter(logging.Formatter(fmt or DEFAULT_LOG_FORMAT))
103    handler.setLevel(numeric)                 # console stays at the requested level
104    root.addHandler(handler)
105
106    # In-memory ring for the "View logs" UI. Capture INFO+ regardless of the
107    # console level (so the log view is useful even at --log-level warning); the
108    # singleton persists its buffer across repeated setup_logging() calls.
109    ring = log_ring()
110    root.addHandler(ring)
111    root.setLevel(min(numeric, logging.INFO))  # let INFO+ reach the handlers

114class TelemetryRecorder:
115    """Keep recent telemetry snapshots in memory and optionally on disk.
116
117    Each snapshot is a plain ``dict`` (typically ``NavigationState.to_dict()``).
118    Calling :meth:`record` always appends to an in-memory ring buffer of at most
119    ``ring_size`` entries, and -- when a ``path`` is configured and the file is
120    open -- also writes the snapshot as one JSON line (JSONL).
121
122    The recorder is usable with ``path=None`` for memory-only operation, and is
123    a no-op-safe context-manager-free design: call :meth:`start` to open the
124    file, :meth:`stop`/:meth:`close` to flush and close it.
125    """
126
127    def __init__(self, path: str | Path | None = None, ring_size: int = 600) -> None:
128        self.path: Path | None = Path(path) if path is not None else None
129        self.ring_size = ring_size
130        self._ring: Deque[dict] = deque(maxlen=ring_size)
131        self._file: TextIO | None = None
132
133    def start(self) -> None:
134        """Open the backing file for appending (no-op when memory-only)."""
135        if self.path is None or self._file is not None:
136            return
137        try:
138            self.path.parent.mkdir(parents=True, exist_ok=True)
139            self._file = self.path.open("a", encoding="utf-8")
140        except OSError:  # pragma: no cover - defensive
141            logger.exception("could not open telemetry file %s", self.path)
142            self._file = None
143
144    def record(self, snapshot: dict) -> None:
145        """Append ``snapshot`` to the ring buffer and, if open, to the file."""
146        self._ring.append(snapshot)
147        if self._file is None:
148            return
149        try:
150            self._file.write(json.dumps(snapshot, default=str) + "\n")
151            self._file.flush()
152        except (OSError, TypeError, ValueError):  # pragma: no cover - defensive
153            logger.exception("could not write telemetry snapshot")
154
155    def recent(self, n: int = 50) -> list[dict]:
156        """Return up to the last ``n`` recorded snapshots (oldest first)."""
157        if n <= 0:
158            return []
159        items = list(self._ring)
160        return items[-n:]
161
162    def stop(self) -> None:
163        """Flush and close the backing file (no-op when memory-only)."""
164        self.close()
165
166    def close(self) -> None:
167        """Close the backing file; the ring buffer is left intact."""
168        if self._file is None:
169            return
170        try:
171            self._file.flush()
172            self._file.close()
173        except OSError:  # pragma: no cover - defensive
174            logger.exception("could not close telemetry file")
175        finally:
176            self._file = None
177
178    def __len__(self) -> int:
179        return len(self._ring)
180

182def wiretap(bus: EventBus, logger: logging.Logger | None = None) -> None:
183    """Log every event published on ``bus`` at DEBUG level.
184
185    Attaches a wildcard subscriber via :meth:`EventBus.subscribe_all`. Cheap
186    when DEBUG is disabled because the handler short-circuits before formatting
187    the payload.
188    """
189    log = logger or logging.getLogger("vanchor.wiretap")
190
191    def _tap(topic: str, payload: Any) -> None:
192        if log.isEnabledFor(logging.DEBUG):
193            log.debug("event %s -> %r", topic, payload)
194
195    bus.subscribe_all(_tap)
